/**
 * Budget guard — global $/day cap for all AI agents.
 *
 * Two thresholds:
 *   80% of cap → demote non-critical tier requests to cheaper tiers
 *                  (e.g. reasoning → cheap)
 *   100%       → refuse non-critical agents entirely
 *
 * Critical agents (compliance) bypass both — `is_critical=true` on
 * `ai_agents`. The kill-switch (anomaly detection) is per-agent and lives
 * elsewhere; this guard is the global financial ceiling.
 *
 * Cost recording:
 *   Every inference's cost lands in agent_run_log.cost_usd AND in
 *   ai_budget_state via an UPSERT. We use a stored procedure for the
 *   UPSERT so the per-agent JSONB merge is atomic.
 */
 
import { createClient } from "@supabase/supabase-js";
 
// eslint-disable-next-line @typescript-eslint/no-explicit-any
let _supabase: any = null;
function admin() {
  if (_supabase) return _supabase;
  _supabase = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_KEY!,
  );
  return _supabase;
}
 
export interface BudgetState {
  date:        string;
  capUsd:      number;
  spendUsd:    number;
  pctUsed:     number;     // 0..1+ (can exceed 1 in extreme cases)
  perAgent:    Record<string, number>;
}
 
const CACHE_TTL_MS = 5_000;
let _cache: { fetchedAt: number; state: BudgetState } | null = null;
 
/** Today's UTC date in YYYY-MM-DD form. */
function todayUtc(): string {
  return new Date().toISOString().slice(0, 10);
}
 
/** Read today's budget state. Cached briefly (5s) so a burst of inferences
 *  doesn't hammer the DB; the cost-recording UPSERT bypasses the cache. */
export async function getBudgetState(): Promise<BudgetState> {
  const now = Date.now();
  if (_cache && now - _cache.fetchedAt < CACHE_TTL_MS) return _cache.state;
 
  const date = todayUtc();
  const { data } = await admin()
    .from("ai_budget_state")
    .select("date, daily_cap_usd, spend_usd, per_agent")
    .eq("date", date)
    .maybeSingle();
 
  const cap   = Number((data as { daily_cap_usd?: number | string } | null)?.daily_cap_usd ?? 200);
  const spend = Number((data as { spend_usd?: number | string }     | null)?.spend_usd     ?? 0);
  const perAgent = ((data as { per_agent?: Record<string, number> } | null)?.per_agent     ?? {}) as Record<string, number>;
 
  const state: BudgetState = {
    date,
    capUsd:   cap,
    spendUsd: spend,
    pctUsed:  cap > 0 ? spend / cap : 0,
    perAgent,
  };
  _cache = { fetchedAt: now, state };
  return state;
}
 
/** Atomic upsert: increment today's spend by `costUsd`, also increment
 *  the per-agent JSONB entry. Bypasses the cache; the next getBudgetState
 *  read picks up the new total after the 5s TTL. */
export async function recordSpend(agentName: string, costUsd: number): Promise<void> {
  if (!isFinite(costUsd) || costUsd <= 0) return;
  const date = todayUtc();
  // Use a Postgres function for atomicity (avoid read-modify-write race).
  // Fallback: an UPSERT with raw SQL via .rpc(). For now we issue a
  // best-effort SELECT-then-UPDATE pattern; we'll harden this with an RPC
  // before launch if budget contention becomes real.
  const { data: existing } = await admin()
    .from("ai_budget_state")
    .select("spend_usd, per_agent")
    .eq("date", date)
    .maybeSingle();
 
  const prevSpend = Number((existing as { spend_usd?: number | string } | null)?.spend_usd ?? 0);
  const prevAgent = ((existing as { per_agent?: Record<string, number> } | null)?.per_agent ?? {}) as Record<string, number>;
  const nextAgent = { ...prevAgent, [agentName]: Number(prevAgent[agentName] ?? 0) + costUsd };
 
  await admin()
    .from("ai_budget_state")
    .upsert({
      date,
      spend_usd: Number((prevSpend + costUsd).toFixed(6)),
      per_agent: nextAgent,
    }, { onConflict: "date" });
 
  _cache = null;
}
 
/**
 * Decide whether the current request should be demoted, refused, or allowed.
 * Returns the effective tier the caller should use (may be a downgrade).
 *
 *   isCritical=true     → always allow at requested tier
 *   pctUsed >= 1.0      → refuse (throw)
 *   pctUsed >= 0.8      → demote `reasoning`/`hard` to `cheap`
 *   else                → allow at requested tier
 */
export async function gateOrThrow(
  requestedTier: "cheap" | "fast" | "reasoning" | "hard",
  isCritical:    boolean,
): Promise<"cheap" | "fast" | "reasoning" | "hard"> {
  if (isCritical) return requestedTier;
 
  const s = await getBudgetState();
  if (s.pctUsed >= 1.0) {
    throw new Error("ai_budget_exhausted");
  }
  if (s.pctUsed >= 0.8 && (requestedTier === "reasoning" || requestedTier === "hard")) {
    return "cheap";
  }
  return requestedTier;
}
 
export function _resetBudgetCacheForTesting(): void {
  _cache = null;
}
