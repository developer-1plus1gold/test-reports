/**
 * Atomically increment a rate-limit counter and check whether the caller
 * is within the allowed budget.
 *
 * Behaviour:
 *  - DB RPC OK + within budget → `{ limited: false }`
 *  - DB RPC OK + over budget   → `{ limited: true, retryAfterSec }`
 *  - DB RPC error/throws       → in-memory fallback at 1/10th budget,
 *                                 admin alert fired (rate-limited to 1/5min)
 *
 * @param key           Unique key (e.g. "otp:+6591234567", "admin-auth:1.2.3.4")
 * @param maxRequests   Maximum requests allowed in the window
 * @param windowMs      Window duration in milliseconds
 */
export async function checkRateLimit(
  key:         string,
  maxRequests: number,
  windowMs:    number,
): Promise<{ limited: boolean; retryAfterSec: number }> {
  const windowSeconds = Math.ceil(windowMs / 1000);
  try {

      p_key:            key,
      p_max_count:      maxRequests,
      p_window_seconds: windowSeconds,
    });
    if (error) {
      console.error("[rate-limit] RPC error (fallback engaged):", error.message);
      fireFallbackAlertOnce(`RPC error: ${error.message}`);
      return checkFallback(key, maxRequests, windowMs);
    }
    const count = data as number;
    if (count > maxRequests) {
      return { limited: true, retryAfterSec: windowSeconds };
    }
    return { limited: false, retryAfterSec: 0 };
  } catch (err) {
    const msg = err instanceof Error ? err.message : String(err);
    console.error("[rate-limit] unexpected error (fallback engaged):", msg);
    fireFallbackAlertOnce(`unexpected error: ${msg}`);
    return checkFallback(key, maxRequests, windowMs);
  }
}

/**
 * Derive a short fingerprint from stable, non-identifying request headers.
 * Used as a last-resort bucket key when no IP and no userId is available.
 * Not cryptographically secure — only needs to differentiate clients.
 */
function fingerprintHeaders(req: Request): string {
  const ua   = req.headers.get("user-agent")      ?? "";
  const lang = req.headers.get("accept-language") ?? "";
  const raw  = `${ua}|${lang}`;
  // djb2 hash — fast, deterministic, no imports needed
  let h = 5381;
  for (let i = 0; i < raw.length; i++) {
    h = Math.imul(h, 33) ^ raw.charCodeAt(i);
  }
  return (h >>> 0).toString(36).padStart(6, "0").slice(-6);
}
