 * admin financial ledger.
 *
 * Categories are a single canonical chart-of-accounts that maps to both
 * SG (Form P partnership) and US (Schedule C) tax lines. Admin picks one
 * canonical category per expense; the SG and US lines are derived
 * automatically and stored on the row, but can be overridden if a specific
 * expense doesn't fit the default mapping.
 *
 * Inventory COGS lives on inventory.cost_per_gram and is intentionally
 * NOT modeled here — it flows to P&L via the inventory→sales path.
 */

// ── FX snapshot ───────────────────────────────────────────────────────────────
 
export interface FxRates {
  /** 1 unit of `currency` = N SGD */
  fxToSgd: number;
  /** 1 unit of `currency` = N USD */
  fxToUsd: number;
  /** Source identifier for traceability */
  source: string;
}
 
const PRICE_SERVICE_URL = process.env.PRICE_SERVICE_URL ?? "http://localhost:3001";
 
/**
 * Fetch live FX rates from the price service (FastForex-backed) and compute
 * conversion factors from `currency` → SGD and `currency` → USD.
 *
 * The price service publishes USD-base rates: `usdRates[X]` = how many X
 * one USD buys. So:
 *   1 currency = (1 / usdRates[currency]) USD
 *              = (usdRates.SGD / usdRates[currency]) SGD
 *
 * Falls back to identity (1:1) if the service is unreachable — admin can
 * always override the SGD/USD amounts manually if the snapshot is wrong.
 */
export async function fetchFxRates(currency: string): Promise<FxRates> {
  const cur = currency.toUpperCase();
  if (cur === "SGD") {
    try {
      const res = await fetch(`${PRICE_SERVICE_URL}/price`, {
        cache: "no-store",
        signal: AbortSignal.timeout(3_000),
      });
      if (res.ok) {
        const data = await res.json();
        const rateUsdPerSgd = data?.usdRates?.SGD;
        if (typeof rateUsdPerSgd === "number" && rateUsdPerSgd > 0) {
          return { fxToSgd: 1, fxToUsd: 1 / rateUsdPerSgd, source: "fastforex" };
        }
      }
    } catch { /* fall through */ }
    return { fxToSgd: 1, fxToUsd: 0.74, source: "fallback" };
  }
 
  if (cur === "USD") {
    try {
      const res = await fetch(`${PRICE_SERVICE_URL}/price`, {
        cache: "no-store",
        signal: AbortSignal.timeout(3_000),
      });
      if (res.ok) {
        const data = await res.json();
        const rateSgdPerUsd = data?.usdRates?.SGD;
        if (typeof rateSgdPerUsd === "number" && rateSgdPerUsd > 0) {
          return { fxToSgd: rateSgdPerUsd, fxToUsd: 1, source: "fastforex" };
        }
      }
    } catch { /* fall through */ }
    return { fxToSgd: 1.35, fxToUsd: 1, source: "fallback" };
  }
 
  // Other currency — convert via USD
  try {
    const res = await fetch(`${PRICE_SERVICE_URL}/price`, {
      cache: "no-store",
      signal: AbortSignal.timeout(3_000),
    });
    if (res.ok) {
      const data = await res.json();
      const rates = data?.usdRates;
      const perUsd = rates?.[cur];
      const sgdPerUsd = rates?.SGD;
      if (typeof perUsd === "number" && perUsd > 0 && typeof sgdPerUsd === "number" && sgdPerUsd > 0) {
        const usdPerUnit = 1 / perUsd;
        return {
          fxToSgd: usdPerUnit * sgdPerUsd,
          fxToUsd: usdPerUnit,
          source:  "fastforex",
        };
      }
    }
  } catch { /* fall through */ }
  return { fxToSgd: 0, fxToUsd: 0, source: "unavailable" };
}

export const ENTRY_TYPES: LedgerEntryType[] = [
  "expense", "capital_in", "owner_draw", "loan_in", "loan_repayment",
];
 
export const EXPENSE_CATEGORIES: ExpenseCategory[] = [
  "software_subscriptions", "marketing_advertising", "professional_services",
  "payment_processing_fees", "shipping_logistics", "office_supplies",
  "rent", "utilities", "telephone_internet",
  "travel", "meals_entertainment",

  "kyc_compliance_tools", "industry_dues",
  "wages_salaries", "depreciation", "other",
];
 
export function isValidEntryType(v: unknown): v is LedgerEntryType {
  return typeof v === "string" && (ENTRY_TYPES as string[]).includes(v);
}
 
export function isValidExpenseCategory(v: unknown): v is ExpenseCategory {
  return typeof v === "string" && (EXPENSE_CATEGORIES as string[]).includes(v);
}
