 *
 * Spec source of truth: https://api-docs.ninjavan.co/static/media/orderapi.eeb602efc469008e7758.yaml
 * (OpenAPI 3.0.1, version 4.1.0 — Order API 4.2 / Cancel 2.2 / Waybill 2.0)
 *
 * NinjaVan does not publish a Node SDK. This module is plain fetch + HMAC,
 * mirroring the structure of src/lib/lalamove.ts.
 *
 * Sandbox traffic ALWAYS goes to https://api-sandbox.ninjavan.co/sg
 * regardless of which country code path you'd use in production.
 *
 * The webhook HMAC key is the same Client Secret used for OAuth.
 */

// ─── Environment / base URL ───────────────────────────────────────────────────
 
function getEnv(): "sandbox" | "production" {
  return ((process.env.NINJAVAN_ENV ?? "sandbox").trim() === "production")
    ? "production"
    : "sandbox";
}
 
function getBaseUrl(): string {
  // Sandbox: docs are explicit that all sandbox requests use lowercase /sg
  // Production: country-coded path (we only ship SG today)
  return getEnv() === "production"
    ? "https://api.ninjavan.co/SG"
    : "https://api-sandbox.ninjavan.co/sg";
}
 
function getCredentials(): { clientId: string; clientSecret: string } {
  const clientId     = process.env.NINJAVAN_CLIENT_ID?.trim();
  const clientSecret = process.env.NINJAVAN_CLIENT_SECRET?.trim();
  if (!clientId || !clientSecret) {
    throw new Error("NINJAVAN_CLIENT_ID / NINJAVAN_CLIENT_SECRET not configured");
  }
  return { clientId, clientSecret };
}
 
// ─── OAuth token cache (module-scoped, in-memory) ─────────────────────────────
//
// Per spec: tokens have a minimum 1-hour TTL. Refresh policy:
//   - 5 minutes before stated expiry, OR
//   - immediately on a 401 from any API call.
 
interface CachedToken {
  accessToken: string;
  expiresAtMs: number;  // epoch ms when token expires
}
 
let tokenCache: CachedToken | null = null;
const REFRESH_BUFFER_MS = 5 * 60 * 1000;
 
// Optional file-based persistence for the token cache. Set
// `NINJAVAN_TOKEN_CACHE_FILE=/tmp/...` in dev/Codespaces so multiple
// short-lived `tsx` script invocations share one token (otherwise
// each invocation = new process = new OAuth call = burns through
// NV's "tokens generated under client_id" audit counter).
//
// In production on Vercel this stays unset — Lambda filesystems are
// ephemeral + per-instance, so file cache provides no benefit and
// can introduce write-contention. The proper prod fix is a shared
// store (Supabase row / Vercel KV) — tracked as a post-launch item.
function getTokenCachePath(): string | null {
  return process.env.NINJAVAN_TOKEN_CACHE_FILE || null;
}
 
function readTokenFromDisk(): CachedToken | null {
  const path = getTokenCachePath();
  if (!path) return null;
  try {
    const fs = require("fs") as typeof import("fs");
    if (!fs.existsSync(path)) return null;
    const raw = fs.readFileSync(path, "utf-8");
    const parsed = JSON.parse(raw) as CachedToken;
    if (typeof parsed.accessToken !== "string" || typeof parsed.expiresAtMs !== "number") return null;
    return parsed;
  } catch { return null; }
}
 
function writeTokenToDisk(t: CachedToken): void {
  const path = getTokenCachePath();
  if (!path) return;
  try {
    const fs = require("fs") as typeof import("fs");
    // Atomic write: tmp + rename so concurrent readers never see partial JSON.
    const tmp = `${path}.${process.pid}.tmp`;
    fs.writeFileSync(tmp, JSON.stringify(t), { mode: 0o600 });
    fs.renameSync(tmp, path);
  } catch {
    // Disk-cache failures are non-fatal — the in-memory cache still works
    // for this process. Worst case the next process fetches a fresh token.
  }
}
 
async function fetchNewToken(): Promise<CachedToken> {
  const { clientId, clientSecret } = getCredentials();
  const res = await fetch(`${getBaseUrl()}/2.0/oauth/access_token`, {
    method:  "POST",
    headers: { "Content-Type": "application/json" },
    body:    JSON.stringify({
      client_id:     clientId,
      client_secret: clientSecret,
      grant_type:    "client_credentials",
    }),
  });
  if (!res.ok) {
    const text = await res.text();
    throw new Error(`NinjaVan OAuth ${res.status}: ${text.slice(0, 200)}`);
  }
  const data = await res.json() as {
    access_token: string;
    expires:      number;     // absolute epoch SECONDS
    expires_in:   number;     // relative seconds
    token_type:   "bearer";
  };
  // Prefer the absolute timestamp (more accurate across clock skew of long-lived caches).
  const expiresAtMs = data.expires
    ? data.expires * 1000
    : Date.now() + (data.expires_in * 1000);
  return { accessToken: data.access_token, expiresAtMs };
}
 
async function getAccessToken(forceRefresh = false): Promise<string> {
  // Tier 1 — in-memory cache (per-process, fastest path).
  if (
    !forceRefresh
    && tokenCache
    && Date.now() < tokenCache.expiresAtMs - REFRESH_BUFFER_MS
  ) {
    return tokenCache.accessToken;
  }
  // Tier 2 — file-backed cache (when NINJAVAN_TOKEN_CACHE_FILE is set).
  // Lets multiple short-lived dev/Codespaces processes share one token
  // so we don't burn NV's per-client_id audit counter on every script run.
  if (!forceRefresh) {
    const onDisk = readTokenFromDisk();
    if (onDisk && Date.now() < onDisk.expiresAtMs - REFRESH_BUFFER_MS) {
      tokenCache = onDisk;
      return onDisk.accessToken;
    }
  }
  // Tier 3 — fetch new + persist to both caches.
  tokenCache = await fetchNewToken();
  writeTokenToDisk(tokenCache);
  return tokenCache.accessToken;
}
 
// ─── REST wrapper ─────────────────────────────────────────────────────────────
//
// Retries once on 401 with a forced token refresh — covers the case where the
// server-side token expired between our check and the request hitting the API.
 
interface RequestOpts {
  method:        "GET" | "POST" | "DELETE";
  path:          string;                    // e.g. "/4.2/orders" — base URL prepended
  body?:         Record<string, unknown>;
  acceptPdf?:    boolean;                   // GET waybill returns binary
}
 
interface RawResponse {
  status:   number;
  buffer:   Buffer;
  contentType: string;
}
 
async function ninjaVanRequest(opts: RequestOpts, retried = false): Promise<RawResponse> {
  const token = await getAccessToken();
  const url   = `${getBaseUrl()}${opts.path}`;
  const res = await fetch(url, {
    method:  opts.method,
    headers: {
      "Authorization": `Bearer ${token}`,
      ...(opts.body  ? { "Content-Type": "application/json" } : {}),
      ...(opts.acceptPdf ? { "Accept": "application/pdf" } : { "Accept": "application/json" }),
    },
    ...(opts.body ? { body: JSON.stringify(opts.body) } : {}),
  });
 
  if (res.status === 401 && !retried) {
    // Token might have just expired server-side — force-refresh and retry once.
    // Also wipe any on-disk cache so peer processes don't keep reading the
    // bad token (only relevant when NINJAVAN_TOKEN_CACHE_FILE is set).
    tokenCache = null;
    const cachePath = process.env.NINJAVAN_TOKEN_CACHE_FILE;
    if (cachePath) {
      try {
        const fs = require("fs") as typeof import("fs");
        if (fs.existsSync(cachePath)) fs.unlinkSync(cachePath);
      } catch { /* best-effort */ }
    }
    return ninjaVanRequest(opts, true);
  }
 
  const arrayBuf = await res.arrayBuffer();
  return {
    status:      res.status,
    buffer:      Buffer.from(arrayBuf),
    contentType: res.headers.get("content-type") ?? "",
  };
}
 
async function ninjaVanJson<T>(opts: RequestOpts): Promise<T> {
  const r = await ninjaVanRequest(opts);
  const text = r.buffer.toString("utf-8");
  if (r.status < 200 || r.status >= 300) {
    throw new Error(`NinjaVan ${opts.method} ${opts.path} → ${r.status}: ${text.slice(0, 300)}`);
  }
  return text ? JSON.parse(text) as T : ({} as T);
}

/**
 * Build a deterministic requested_tracking_number from our internal order id.
 *
 * Verified live against sandbox 2026-04-29: when a shipper account has a
 * prefix configured (1Plus1Gold's does), the API rejects:
 *   - any value > 9 chars
 *   - any value containing '-' (the documented permissive pattern is rejected)
 *   - any value containing the prefix itself ("please do not pass a prefix")
 *
 * Effective rule: send 1-9 alphanumeric characters; NinjaVan concatenates
 * the account's configured prefix. We use the first 9 hex chars of the UUID.
 *
 * Deterministic so a retried booking sends the same value — NinjaVan rejects
 * duplicate requested_tracking_numbers, giving us idempotency on transient
 * network failures.
 */
export function buildRequestedTrackingNumber(internalOrderId: string): string {
  const clean = internalOrderId.replace(/[^a-zA-Z0-9]/g, "").slice(0, 9);
  return clean.length >= 1 ? clean : "fallback1";
}

/**
 * Per-day-of-week shipper configuration, distilled from `shipper_schedules`.
 * day-of-week is 0=Sun..6=Sat (matches JS `Date.getUTCDay()`).
 */
export interface ShipperDayConfig {
  enabled:        boolean;
  /** Hour of day in SGT after which same-day pickup is no longer eligible. null = no cutoff (same-day all day). */
  cutoffHourSgt:  number | null;
}
 
function formatSgtDate(d: Date): string {
  return `${d.getUTCFullYear()}-${String(d.getUTCMonth() + 1).padStart(2, "0")}-${String(d.getUTCDate()).padStart(2, "0")}`;
}
 
/**
 * Compute the next eligible pickup date based on the per-day-of-week schedule
 * configured in `shipper_schedules` (admin Settings → NinjaVan).
 *
 * Logic:
 *   - Today: if `enabled` AND now is before today's `cutoffHourSgt` → today
 *   - Otherwise advance day-by-day until we find an `enabled` day
 *   - Returns yyyy-MM-dd in Singapore time
 *
 * Holiday handling: NinjaVan auto-adjusts blocked dates (public holidays)
 * server-side and echoes the actual chosen date in the response — we don't
 * model the SG holiday calendar locally.
 */
export function computePickupDate(daysByDow: Map<number, ShipperDayConfig>, now: Date = new Date()): string {
  // SGT is UTC+8 with no DST — shift by a fixed +8h to get Singapore-local view.
  const sgt = new Date(now.getTime() + 8 * 60 * 60 * 1000);
  const sgtHour = sgt.getUTCHours();
  let cursor = new Date(Date.UTC(sgt.getUTCFullYear(), sgt.getUTCMonth(), sgt.getUTCDate()));
 
  const today = daysByDow.get(cursor.getUTCDay());
  if (today?.enabled && (today.cutoffHourSgt === null || sgtHour < today.cutoffHourSgt)) {
    return formatSgtDate(cursor);
  }
 
  // Advance one day at a time until we find an enabled day. Capped at 14
  // iterations as a safety belt — a misconfigured schedule with all days
  // disabled would otherwise loop forever.
  for (let i = 0; i < 14; i++) {
    cursor = new Date(Date.UTC(cursor.getUTCFullYear(), cursor.getUTCMonth(), cursor.getUTCDate() + 1));
    const cfg = daysByDow.get(cursor.getUTCDay());
    if (cfg?.enabled) return formatSgtDate(cursor);
  }
 
  // Sanity fallback — only reachable if every day is disabled. Returning
  // today's date lets NinjaVan reject the booking with a clear error rather
  // than us silently shipping an arbitrary date.
  const sgtToday = new Date(Date.UTC(sgt.getUTCFullYear(), sgt.getUTCMonth(), sgt.getUTCDate()));
  return formatSgtDate(sgtToday);
}
 
/**
 * Schedule-free fallback: same logic as `computePickupDate` but with a
 * hardcoded weekday-only schedule and a default cutoff. Kept for tests and
 * for callers that don't have access to `shipper_schedules` (none today).
 */
export function computeNextBusinessDate(now: Date = new Date(), cutoffHourSGT = 14): string {
  const weekdaysOnly = new Map<number, ShipperDayConfig>([
    [1, { enabled: true,  cutoffHourSgt: cutoffHourSGT }],
    [2, { enabled: true,  cutoffHourSgt: cutoffHourSGT }],
    [3, { enabled: true,  cutoffHourSgt: cutoffHourSGT }],
    [4, { enabled: true,  cutoffHourSgt: cutoffHourSGT }],
    [5, { enabled: true,  cutoffHourSgt: cutoffHourSGT }],
    [0, { enabled: false, cutoffHourSgt: null }],
    [6, { enabled: false, cutoffHourSgt: null }],
  ]);
  return computePickupDate(weekdaysOnly, now);
}

export async function createOrder(input: CreateOrderInput): Promise<CreateOrderResult> {
  const requested = buildRequestedTrackingNumber(input.internalOrderId);
 
  // ParcelJob4.2: required = dimensions, delivery_start_date, delivery_timeslot, is_pickup_required
  const body = {
    service_type:  "Parcel",
    // Default Nextday — every 1Plus1Gold shipment goes out as next-business-
    // day (Lalamove covers same-day; NinjaVan is the next-day fallback).
    // Tells NV's dashboard this is a next-day shipment instead of bucketing
    // as generic Standard. Audit/test paths can override via
    // input.serviceLevel — Test Scenario #1 of the production-approval
    // audit requires "Standard" specifically. Valid values per spec:
    // Standard | Express | Sameday | Nextday.
    service_level: input.serviceLevel ?? "Nextday",
    requested_tracking_number: requested,
    reference: {
      // Use the same 8-char short ref the admin UI shows ("#6D4220A1") so
      // the printed waybill stays scannable to a person and matches what
      // staff see on screen. The full UUID is never printed.
      merchant_order_number: input.internalOrderId.replace(/-/g, "").slice(0, 8).toUpperCase(),
    },
    from: pickupContact(),
    to:   input.to,
    parcel_job: {
      is_pickup_required: true,
      // Standing daily pickup reservation: NinjaVan matches on (date, address)
      // — driver collects in the dashboard-configured window. We still pass the
      // full block so a Scheduled fallback works if no standing reservation exists.
      pickup_service_type:  "Scheduled",
      pickup_service_level: "Standard",
      pickup_date:          input.pickupDate,
      pickup_timeslot: {
        // 09:00–18:00 — NinjaVan's standard pickup window. Earlier code
        // had 22:00 to dodge the spec's "pickup end-time must be ≥3h
        // after booking time" rule on late-day bookings, but the
        // integration audit (Test Scenario #1, 2026-05-06) flagged any
        // value other than 18:00 as invalid. The ≥3h rule is satisfied
        // by computeNextBusinessDate's 14:00 SGT cutoff: a 13:59
        // booking against 18:00 leaves 4h+ — bookings after 14:00 roll
        // to next-day pickup with the full 09:00-18:00 window.
        start_time: "09:00",
        end_time:   "18:00",
        timezone:   "Asia/Singapore",
      },
      pickup_approximate_volume: "Less than 3 Parcels",
      delivery_start_date: input.deliveryStartDate,
      delivery_timeslot: {
        start_time: "09:00",
        end_time:   "22:00",
        timezone:   "Asia/Singapore",
      },
      // Neutral wording — never telegraph contents on a label or driver app.
      // "ID required" is operational guidance and is normal for high-value
      // deliveries without revealing what's inside.
      delivery_instructions: input.deliveryInstructions ?? "Handle with care. Recipient ID required at delivery.",
      allow_weekend_delivery: true,
      dimensions: { weight: input.weightKg },
      items: [
        { item_description: input.itemDescription, quantity: 1, is_dangerous_good: false },
      ],
    },
  };
 
  const result = await ninjaVanJson<{
    requested_tracking_number?: string;
    tracking_number:            string;
  }>({ method: "POST", path: "/4.2/orders", body });
 
  return {
    trackingNumber:          result.tracking_number,
    requestedTrackingNumber: result.requested_tracking_number ?? requested,
  };
}

/**
 * Fetch a printable A4 PDF waybill for a tracking number.
 *
 * Per spec, waybill API requires a separate access grant from order creation
 * (audited by NinjaVan even in sandbox). If the grant isn't in place, this
 * call returns 403 — the caller (admin route) maps that to a friendly error.
 */
export async function getWaybillPdf(trackingNo: string): Promise<Buffer> {
  // hide_shipper_details=true: omit company name + contact details from the
  // printed label. OPSEC — keeps "1+1 GOLD" off any label seen in the
  // warehouse or by last-mile drivers. NinjaVan still has the data internally
  // for return-to-sender; only the visible printed label is anonymized.
  const r = await ninjaVanRequest({
    method:    "GET",
    path:      `/2.0/reports/waybill?tid=${encodeURIComponent(trackingNo)}&hide_shipper_details=true&orientation=portrait`,
    acceptPdf: true,
  });
  if (r.status < 200 || r.status >= 300) {
    throw new Error(`NinjaVan waybill ${trackingNo} → ${r.status}: ${r.buffer.toString("utf-8").slice(0, 200)}`);
  }
  if (!r.contentType.toLowerCase().includes("pdf")) {
    throw new Error(`NinjaVan waybill ${trackingNo}: unexpected content-type "${r.contentType}"`);
  }
  return r.buffer;
}
 
// ─── Webhook signature verification ───────────────────────────────────────────
//
// Spec quote (verbatim):
//   "Ninja Van passes along a digital signature with each request in the
//    `X-Ninjavan-Hmac-Sha256` header. The signature is generated with the HMAC
//    algorithm, using your Client Secret as a key and SHA256 digest mode."
//   "The json body used for hash generation should not be encoded/encrypted in any way."
//
// Implication: HMAC must be computed over the EXACT raw request bytes. The
// caller MUST pass `await req.text()`, never `JSON.stringify(await req.json())`
// — re-serialization changes whitespace and key order, breaking the digest.
 
export function verifyWebhookSignature(rawBody: string, signatureHeader: string | null | undefined): boolean {
  if (!signatureHeader) return false;
  const secret = process.env.NINJAVAN_CLIENT_SECRET?.trim();
  if (!secret) return false;
  try {
    const expected = createHmac("sha256", secret).update(rawBody, "utf8").digest();   // 32 bytes
    const actual   = Buffer.from(signatureHeader, "base64");
    if (actual.length !== expected.length) return false;
    return timingSafeEqual(expected, actual);
  } catch {
    return false;
  }
}

const STATUS_FALLBACK_MAP: Record<string, string> = {
  "Pending Pickup":         "booked",
  "Picked Up":              "picked_up",
  "On Vehicle for Delivery": "out_for_delivery",
  "Delivered":              "delivered",

  "Delivery Exception":     "out_for_delivery",
  "Cancelled":              "failed",
  "Returned to Sender":     "failed",
};
 
export function mapNinjaVanEvent(event: string | null | undefined, status: string | null | undefined = null): string {
  if (event && EVENT_MAP[event]) return EVENT_MAP[event];
  if (status && STATUS_FALLBACK_MAP[status]) return STATUS_FALLBACK_MAP[status];
  return "pending";
}
