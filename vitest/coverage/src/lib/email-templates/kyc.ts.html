/**
 * KYC lifecycle emails — mirrors the four WhatsApp templates:
 *   1p1g_kyc_approved   — KYC verified, account unlocked
 *   1p1g_kyc_rejected   — KYC rejected, retry or contact us
 *   1p1g_kyc_on_hold_v1 — manual review needed
 *   1p1g_kyc_expiring_v1 — verification expires soon, please re-verify
 *
 * All four exported as separate functions sharing the same brand layout.
 * Tone matches the calm-factual style that got the WhatsApp templates
 * Meta-approved on first try (see feedback_meta_template_utility_category.md).
 */
import { renderEmail, EMAIL_COLORS, EMAIL_FONTS } from "./layout";
 
type Base = { firstName?: string };
 
export function kycApprovedEmail(data: Base): { subject: string; html: string; text: string } {
  const greeting = data.firstName ? `Hi ${data.firstName},` : "Hi,";
  const content = `
    <p style="margin: 0 0 16px;">${escapeHtml(greeting)}</p>
    <p style="margin: 0 0 16px;">Your identity verification is complete. Your account is now fully active.</p>
    <p style="margin: 0; font-family: ${EMAIL_FONTS.sans}; font-size: 13px; color: ${EMAIL_COLORS.muted};">
      You can place orders of any value, including those above S$5,000.
    </p>
  `;
  return {
    subject: "Identity verification complete",
    html: renderEmail({ preheader: "Your KYC is approved — account fully active.", content, cta: { label: "Visit your account →", url: "https://www.1plus1gold.com/account" } }),
    text: [greeting, "", "Your identity verification is complete. Your account is now fully active.", "", "Visit: https://www.1plus1gold.com/account", "", "1Plus1Gold · Singapore"].join("\n"),
  };
}
 
export function kycRejectedEmail(data: Base & { reasonLabel?: string }): { subject: string; html: string; text: string } {
  const greeting = data.firstName ? `Hi ${data.firstName},` : "Hi,";
  const reason   = data.reasonLabel ? `Reason: ${data.reasonLabel}` : "We were unable to verify your identity from the documents provided.";
  const content = `
    <p style="margin: 0 0 16px;">${escapeHtml(greeting)}</p>
    <p style="margin: 0 0 16px;">${escapeHtml(reason)}</p>
    <p style="margin: 0 0 16px;">You can retry verification anytime — most cases are resolved by re-uploading clearer photos of your ID.</p>
    <p style="margin: 0; font-family: ${EMAIL_FONTS.sans}; font-size: 13px; color: ${EMAIL_COLORS.muted};">
      Need help? Reply to this email or visit our contact page.
    </p>
  `;
  return {
    subject: "Identity verification — action needed",
    html: renderEmail({ preheader: "Your KYC could not be verified — please retry.", content, cta: { label: "Retry verification →", url: "https://www.1plus1gold.com/account?kyc=retry" } }),
    text: [greeting, "", reason, "", "You can retry verification anytime.", "", "Retry: https://www.1plus1gold.com/account?kyc=retry", "", "1Plus1Gold · Singapore"].join("\n"),
  };
}
 
export function kycOnHoldEmail(data: Base & { reasonLabel?: string }): { subject: string; html: string; text: string } {
  const greeting = data.firstName ? `Hi ${data.firstName},` : "Hi,";
  const reason   = data.reasonLabel ? `Reason: ${data.reasonLabel}` : "Your verification needs a quick manual review by our compliance team.";
  const content = `
    <p style="margin: 0 0 16px;">${escapeHtml(greeting)}</p>
    <p style="margin: 0 0 16px;">${escapeHtml(reason)}</p>
    <p style="margin: 0; font-family: ${EMAIL_FONTS.sans}; font-size: 13px; color: ${EMAIL_COLORS.muted};">
      This usually takes one business day. We&apos;ll email you the moment it&apos;s resolved.
    </p>
  `;
  return {
    subject: "Identity verification — on hold",
    html: renderEmail({ preheader: "Your KYC is on hold pending manual review.", content }),
    text: [greeting, "", reason, "", "This usually takes one business day.", "", "1Plus1Gold · Singapore"].join("\n"),
  };
}
 
export function kycExpiringEmail(data: Base & { daysRemaining: number }): { subject: string; html: string; text: string } {
  const greeting = data.firstName ? `Hi ${data.firstName},` : "Hi,";
  const window   = data.daysRemaining <= 1 ? "tomorrow" : `in ${data.daysRemaining} days`;
  const content = `
    <p style="margin: 0 0 16px;">${escapeHtml(greeting)}</p>
    <p style="margin: 0 0 16px;">Your identity verification expires ${escapeHtml(window)}.</p>
    <p style="margin: 0; font-family: ${EMAIL_FONTS.sans}; font-size: 13px; color: ${EMAIL_COLORS.muted};">
      Re-verifying takes about 2 minutes — refresh now to avoid interruption on your next order.
    </p>
  `;
  return {
    subject: `Identity verification expires ${data.daysRemaining <= 1 ? "tomorrow" : `in ${data.daysRemaining} days`}`,
    html: renderEmail({ preheader: `Your KYC expires ${window}. Re-verify now.`, content, cta: { label: "Re-verify identity →", url: "https://www.1plus1gold.com/account?kyc=renew" } }),
    text: [greeting, "", `Your identity verification expires ${window}.`, "Re-verifying takes about 2 minutes.", "", "Re-verify: https://www.1plus1gold.com/account?kyc=renew", "", "1Plus1Gold · Singapore"].join("\n"),
  };
}
 
function escapeHtml(s: string): string {
  return s.replace(/&/g, "&amp;").replace(/</g, "&lt;").replace(/>/g, "&gt;").replace(/"/g, "&quot;").replace(/'/g, "&#39;");
}
