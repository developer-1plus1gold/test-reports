/**
 * P&L aggregation — pure logic, unit-testable without DB.
 *
 * Revenue lines (cash basis — uses created_at + paid status):
 *   - Gold sales       — orders.subtotal_ipm_sgd       for paid orders
 *   - Accessories      — orders.subtotal_taxable_sgd   for paid orders
 *   - Conversion fees  — conversions.conversion_fee_sgd for paid conversions
 *   - Shipping         — orders.shipping_fee_sgd       (charged to customer)
 *   - Processing       — orders.processing_fee_sgd     (charged to customer)
 *
 * COGS — TRUE landed cost from procurement, not spot proxy:
 *   - Gold COGS = sum(inventory.cost_per_g_sgd × products.grams) for inventory
 *     rows that flipped to status='sold' inside the period. cost_per_g_sgd is
 *     the landed cost (gold + PO-side shipping + payment + insurance + other,
 *     allocated across the lines in the receipt). PO ancillary costs flow
 *     through automatically.
 *   - Accessory COGS = sum(qty × latest_landed_cost_sgd) per product. Latest
 *     PO line landed cost is the cost-basis proxy (accessories aren't tracked
 *     per-unit in the inventory table; they live in products.stock_qty).
 *
 * Operating expenses:
 *   - From ledger_entries where entry_type='expense' AND entry_date in period.
 *   - Grouped by tax category for the active jurisdiction (SG / US).
 *   - Note: PO-side costs (shipping/payment/insurance) are NOT in ledger
 *     entries — they're inside landed_cost and flow through COGS instead.
 *     Sales-side outflows (Lalamove invoice, gateway %) ARE expected as
 *     ledger entries (shipping_logistics + payment_processing_fees).
 */

export function resolvePeriod(period: Period, at: Date): {
  start: string;
  end:   string;
  label: string;
} {
  const y = at.getUTCFullYear();
  const m = at.getUTCMonth();
 
  if (period === "month") {
    const start = new Date(Date.UTC(y, m,   1));
    const end   = new Date(Date.UTC(y, m+1, 0));
    return { start: iso(start), end: iso(end), label: monthLabel(start) };
  }
 
  if (period === "quarter") {
    const qIdx = Math.floor(m / 3);
    const start = new Date(Date.UTC(y, qIdx * 3,     1));
    const end   = new Date(Date.UTC(y, qIdx * 3 + 3, 0));
    return { start: iso(start), end: iso(end), label: `Q${qIdx + 1} ${y}` };
  }
 
  const start = new Date(Date.UTC(y, 0,  1));
  const end   = new Date(Date.UTC(y, 11, 31));
  return { start: iso(start), end: iso(end), label: `FY ${y}` };
}
 
export function stepPeriod(period: Period, anchor: Date, dir: 1 | -1): Date {
  const y = anchor.getUTCFullYear();
  const m = anchor.getUTCMonth();
  const d = anchor.getUTCDate();
  if (period === "month")   return new Date(Date.UTC(y, m + dir, Math.min(d, 28)));
  if (period === "quarter") return new Date(Date.UTC(y, m + dir * 3, Math.min(d, 28)));
  return new Date(Date.UTC(y + dir, m, Math.min(d, 28)));
}

export function buildPnlReport(args: {

  at:              Date;
  jurisdiction:    Jurisdiction;
  orders:          PaidOrderRow[];
  conversions:     PaidConversionRow[];
  expenses:        ExpenseRow[];

  soldInventory:   SoldInventoryRow[];
  /** Per-line accessory cost-basis pairs for orders inside the period. */
  accessoryCogs:   AccessoryCogsRow[];
}): PnlReport {

  const { start, end, label } = resolvePeriod(period, at);
 
  // Revenue.
  const paidOrders = orders.filter(isOrderPaid);
  let goldRev = 0, accRev = 0, shipRev = 0, procRev = 0;
  for (const o of paidOrders) {
    goldRev += Number(o.subtotal_ipm_sgd     ?? 0);
    accRev  += Number(o.subtotal_taxable_sgd ?? 0);
    shipRev += Number(o.shipping_fee_sgd     ?? 0);
    procRev += Number(o.processing_fee_sgd   ?? 0);
  }
 
  let convRev = 0;
  for (const c of conversions) {
    if (!c.paid_at) continue;
    convRev += Number(c.conversion_fee_sgd ?? 0);
  }
 
  const totalRev = goldRev + accRev + convRev + shipRev + procRev;
 
  // COGS — landed cost from inventory (gold + standard) and PO history (accessories).
  let goldCogs = 0;
  for (const inv of soldInventory) {
    goldCogs += Number(inv.cost_per_g_sgd ?? 0) * Number(inv.grams ?? 0);
  }
  let accCogs = 0;
  for (const a of accessoryCogs) {
    accCogs += Number(a.quantity ?? 0) * Number(a.landed_cost_sgd ?? 0);
  }
  const totalCogs   = goldCogs + accCogs;
  const grossProfit = totalRev - totalCogs;
 
  // Expenses grouped by tax line.
  const lineMap = new Map<string, { categories: Set<string>; amount: number }>();
  for (const e of expenses) {
    const taxLine = jurisdiction === "sg" ? e.sg_tax_category : e.us_tax_category;
    if (!taxLine) continue;
    const cur = lineMap.get(taxLine) ?? { categories: new Set(), amount: 0 };
    cur.amount += Number(e.amount_sgd ?? 0);
    if (e.category) cur.categories.add(e.category);
    lineMap.set(taxLine, cur);
  }
  const lines: CategoryLine[] = [...lineMap.entries()]
    .map(([taxLine, v]) => ({
      taxLine,
      categories: [...v.categories].sort(),
      amount_sgd: roundCents(v.amount),
    }))
    .sort((a, b) => b.amount_sgd - a.amount_sgd);
  const totalExpenses = lines.reduce((s, l) => s + l.amount_sgd, 0);
 
  return {
    periodStart:  start,
    periodEnd:    end,
    periodLabel:  label,
    jurisdiction,
    revenue: {
      gold_sgd:        roundCents(goldRev),
      accessories_sgd: roundCents(accRev),
      conversion_sgd:  roundCents(convRev),
      shipping_sgd:    roundCents(shipRev),
      processing_sgd:  roundCents(procRev),
      total_sgd:       roundCents(totalRev),
    },
    cogs: {
      gold_sgd:        roundCents(goldCogs),
      accessories_sgd: roundCents(accCogs),
      total_sgd:       roundCents(totalCogs),
    },
    grossProfit_sgd: roundCents(grossProfit),
    expenses: {
      lines,
      total_sgd: roundCents(totalExpenses),
    },
    netIncome_sgd: roundCents(grossProfit - totalExpenses),
  };
}
