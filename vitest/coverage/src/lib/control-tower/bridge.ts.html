 * Control Tower bridge — RETIRED 2026-05-07 (Phase B complete).
 *
 * Each domain agent (playbook_curator B.1, security_agent B.2,
 * compliance B.3) now writes to `agent_decisions` directly when emitting
 * work. The synthesisers + bridge field on QueueRow are kept here as a
 * historical reference + for the QueueRow type which is still imported
 * by the route handler. If a future legacy integration arises, the
 * pattern follows: write `agent_decisions` alongside the domain row +
 * mirror status changes from the source-of-truth update path.
 */

  // bridge field deprecated post-B.3; still on the type for any cached
  // page state that may reference it. New rows will not set it.
  bridge?:              { table: string; id: string; href: string };
}

/**
 * Re-sort a heterogeneous queue using the same severity rank the API
 * already applies. Kept as a generic helper — useful for any queue that
 * mixes severities even after the bridge retirement.
 */
const SEV_RANK: Record<string, number> = { escalate: 1, approve: 2, notify: 3, auto: 4 };
export function sortQueueBySeverityThenAge(queue: QueueRow[]): QueueRow[] {
  return [...queue].sort((a, b) => {
    const sa = SEV_RANK[a.severity] ?? 99;
    const sb = SEV_RANK[b.severity] ?? 99;
    if (sa !== sb) return sa - sb;
    return b.created_at.localeCompare(a.created_at);
  });
}
