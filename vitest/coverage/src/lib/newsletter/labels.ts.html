/**
 * Customer-visible UI strings on the newsletter (PDF + page).
 *
 * Includes both short labels (section heads, stat labels, footer) AND
 * long-prose templates (cover lead, Premium Curve description, "Other
 * gold stories" bodies, captions, "What's next" nudges). Templates carry
 * {placeholder} markers the renderer substitutes at render time with
 * pre-formatted runtime values (grams, money, percentages, names). The
 * translator preserves {placeholder} markers verbatim — same convention
 * as {{TOKEN_N}} for tokenised numerics.
 *
 * Brand wordmarks ("1PLUS1GOLD", "Premium Curve", "Combibar", "Valcambi")
 * are intentionally excluded from translation. "Gold Yield" is a
 * programme name — explicitly translated so non-English readers
 * understand it (vs left as a brand wordmark).
 *
 * Plural handling: plural English nouns are used throughout
 * ("X items", "X bars", "X times", "X purchases"). Slightly off when
 * count = 1 in English ("1 bars enrolled") — tolerated for v1 in
 * exchange for clean translation paths.
 *
 * Translation flow:
 *   EN_LABELS → tokenise (none — placeholders look like tokens) → translator
 *   → cache in newsletter_translations.translated_template_strings.uiLabels
 *   → renderer reads via mergeLabels() + fmtLabel() at render time.
 */

/**
 * fmt(template, vars) — substitute {placeholder} tokens in a label.
 * Used at render time after the dictionary lookup. Unknown placeholders
 * pass through unchanged so missing translations fail visibly rather
 * than silently inserting empty strings.
 */
export function fmt(template: string, vars: Record<string, string | number>): string {
  return template.replace(/\{(\w+)\}/g, (m, key) => {
    const v = vars[key];
    return v == null ? m : String(v);
  });
}
 
/**
 * Merge a translated subset over the English fallback so renderers can
 * always read every key. Translator may return a partial map (some
 * languages drop some keys); this fills the gaps with English so the
 * page never shows an empty section heading.
 */
export function mergeLabels(translated?: Partial<NewsletterLabels> | null): NewsletterLabels {
  if (!translated) return EN_LABELS;
  return { ...EN_LABELS, ...translated };
}
