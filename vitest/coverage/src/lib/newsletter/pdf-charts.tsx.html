/**
 * PDF chart components for the monthly newsletter.
 *
 * Rendered server-side via @react-pdf/renderer's SVG primitives (no canvas,
 * no external service). Each chart is intentionally small + editorial in
 * style — these support the prose, they don't substitute for a full
 * dashboard. For richer interactive charts, the on-screen page uses
 * Recharts; the PDF mirrors via these static SVGs.
 *
 * Palette aligned with the editorial newsletter rebuild:
 *   gold   #C9A84C — primary accent + structural rules
 *   green  #2E8B57 — gains / accumulation / milestones
 *   red    #C45A4A — losses / outflows
 *   amber  #D69E2E — pending / mid-state / alerts
 *   blue   #2F6FAD — neutral / informational / market data
 *   brown  #5A4A2A — body text, gridlines
 *   subtle #DCD3B8 — gridlines, axis ticks
 */
 
import { Svg, Polyline, Polygon, Line, Path, Circle, Rect, Text, G } from "@react-pdf/renderer";
// niceMilestones lives in chart-utils.ts so the page renderer
// (page-charts.tsx, "use client") can import it without dragging
// @react-pdf/renderer into the client bundle.
import { niceMilestones } from "./chart-utils";
export { niceMilestones };
 
export const CHART_PALETTE = {
  gold:   "#C9A84C",
  green:  "#2E8B57",
  red:    "#C45A4A",
  amber:  "#D69E2E",
  blue:   "#2F6FAD",
  brown:  "#5A4A2A",
  subtle: "#DCD3B8",
  paper:  "#FBF8F0",
};
 
// ── Sparkline ───────────────────────────────────────────────────────────
// Tiny inline trend chart for headlines + tile decoration. Auto-scales to
// the value range; closes the line with a soft fill underneath. Up = green,
// down = red.
export function Sparkline({
  data,
  width  = 120,
  height = 30,
  color,
}: {
  data: number[];
  width?: number;
  height?: number;
  color?: string;
}) {
  if (data.length < 2) return null;
  const min = Math.min(...data);
  const max = Math.max(...data);
  const range = max - min || 1;
  const stepX = width / (data.length - 1);
  const points = data.map((v, i) => {
    const x = i * stepX;
    const y = height - ((v - min) / range) * (height - 4) - 2;
    return `${x.toFixed(2)},${y.toFixed(2)}`;
  });
  const trendUp = data[data.length - 1] >= data[0];
  const stroke = color ?? (trendUp ? CHART_PALETTE.green : CHART_PALETTE.red);
  // Build a closed polygon for the soft area fill underneath.
  const areaPoints = [...points, `${width},${height}`, `0,${height}`].join(" ");
  return (
    <Svg width={width} height={height}>
      <Polygon points={areaPoints} fill={stroke} fillOpacity={0.10} />
      <Polyline points={points.join(" ")} stroke={stroke} strokeWidth={1.5} fill="none" />
      <Circle cx={(data.length - 1) * stepX} cy={height - ((data[data.length - 1] - min) / range) * (height - 4) - 2} r={2} fill={stroke} />
    </Svg>
  );
}
 
// ── Line chart with gridlines + axis labels ────────────────────────────
// Used for the spot price trend on the market page. Larger than a
// sparkline — has gridlines, min/max labels, and a colored end-marker.
export function LineChart({
  data,
  width  = 380,
  height = 140,
  color  = CHART_PALETTE.blue,
  yLabelFmt,
  xLabelFirst,
  xLabelLast,
}: {
  data: number[];
  width?: number;
  height?: number;
  color?: string;
  yLabelFmt?: (v: number) => string;
  xLabelFirst?: string;
  xLabelLast?: string;
}) {
  if (data.length < 2) return null;
  const padL = 36, padR = 8, padT = 10, padB = 22;
  const innerW = width - padL - padR;
  const innerH = height - padT - padB;
  const min = Math.min(...data);
  const max = Math.max(...data);
  const range = max - min || 1;
  const stepX = innerW / (data.length - 1);
  const yOf = (v: number) => padT + (1 - (v - min) / range) * innerH;
  const xOf = (i: number) => padL + i * stepX;
  const fmt = yLabelFmt ?? ((v: number) => v.toFixed(0));
 
  const linePts = data.map((v, i) => `${xOf(i).toFixed(2)},${yOf(v).toFixed(2)}`);
  const areaPts = [...linePts, `${xOf(data.length - 1).toFixed(2)},${padT + innerH}`, `${xOf(0).toFixed(2)},${padT + innerH}`].join(" ");
 
  // Three horizontal gridlines at min / mid / max for visual orientation.
  const mid = (min + max) / 2;
  const gridYs = [
    { y: padT,                v: max },
    { y: padT + innerH / 2,   v: mid },
    { y: padT + innerH,       v: min },
  ];
 
  return (
    <Svg width={width} height={height}>
      {/* Gridlines + y labels */}
      {gridYs.map((g, i) => (
        <G key={i}>
          <Line x1={padL} y1={g.y} x2={width - padR} y2={g.y} stroke={CHART_PALETTE.subtle} strokeWidth={0.5} />
          <Text x={padL - 4} y={g.y + 3} style={{ fontSize: 7, fontFamily: "Montserrat", fill: CHART_PALETTE.brown, textAnchor: "end" }}>
            {fmt(g.v)}
          </Text>
        </G>
      ))}
      {/* Filled area + line */}
      <Polygon points={areaPts} fill={color} fillOpacity={0.10} />
      <Polyline points={linePts.join(" ")} stroke={color} strokeWidth={1.6} fill="none" />
      {/* End marker */}
      <Circle cx={xOf(data.length - 1)} cy={yOf(data[data.length - 1])} r={2.5} fill={color} />
      {/* X labels */}
      {xLabelFirst && (
        <Text x={padL} y={height - 4} style={{ fontSize: 7, fontFamily: "Montserrat", fill: CHART_PALETTE.brown }}>
          {xLabelFirst}
        </Text>
      )}
      {xLabelLast && (
        <Text x={width - padR} y={height - 4} style={{ fontSize: 7, fontFamily: "Montserrat", fill: CHART_PALETTE.brown, textAnchor: "end" }}>
          {xLabelLast}
        </Text>
      )}
    </Svg>
  );
}
 
// ── Accumulation area chart ─────────────────────────────────────────────
// Trailing 12-month cumulative grams as a smooth area chart. Shows the
// customer's accumulation arc — clearly upward-trending if they're an
// active accumulator.
export function AccumulationChart({
  series,
  width  = 380,
  height = 140,
  highlightLastN = 1,
}: {
  series: { yearMonth: string; gramsCumulative: number }[];
  width?: number;
  height?: number;
  highlightLastN?: number;
}) {
  if (series.length < 2) return null;
  const padL = 32, padR = 8, padT = 8, padB = 22;
  const innerW = width - padL - padR;
  const innerH = height - padT - padB;
  const values = series.map(s => s.gramsCumulative);
  const max = Math.max(...values);
  const min = 0; // accumulation is always anchored at 0
  const range = (max - min) || 1;
  const stepX = innerW / (series.length - 1);
  const yOf = (v: number) => padT + (1 - (v - min) / range) * innerH;
  const xOf = (i: number) => padL + i * stepX;
 
  const linePts = series.map((s, i) => `${xOf(i).toFixed(2)},${yOf(s.gramsCumulative).toFixed(2)}`);
  const areaPts = [...linePts, `${xOf(series.length - 1).toFixed(2)},${padT + innerH}`, `${xOf(0).toFixed(2)},${padT + innerH}`].join(" ");
 
  // Adaptive y-axis milestones. For accumulating customers (max ≤ 150g)
  // we keep the editorial 25/50/100 milestone semantic — those are the
  // Premium Curve cycle markers customers think in. For larger
  // lifetimes (Mukesh's 2606g case earlier) all three collapse to <4%
  // of chart height and overlap; switch to "nice tick" intervals
  // (250 / 500 / 1000 / 2000…) so labels stay readable.
  const milestones = niceMilestones(max);
 
  return (
    <Svg width={width} height={height}>
      {/* Milestone gridlines + labels */}
      {milestones.map((m, i) => {
        const y = yOf(m);
        return (
          <G key={i}>
            <Line x1={padL} y1={y} x2={width - padR} y2={y} stroke={CHART_PALETTE.subtle} strokeWidth={0.5} strokeDasharray="2,2" />
            <Text x={padL - 4} y={y + 3} style={{ fontSize: 7, fontFamily: "Montserrat", fill: CHART_PALETTE.brown, textAnchor: "end" }}>
              {m}g
            </Text>
          </G>
        );
      })}
      {/* Area + line */}
      <Polygon points={areaPts} fill={CHART_PALETTE.green} fillOpacity={0.15} />
      <Polyline points={linePts.join(" ")} stroke={CHART_PALETTE.green} strokeWidth={1.8} fill="none" />
      {/* Highlight the most recent N points (current month etc.) */}
      {series.slice(-highlightLastN).map((s, i) => {
        const idx = series.length - highlightLastN + i;
        return <Circle key={i} cx={xOf(idx)} cy={yOf(s.gramsCumulative)} r={2.5} fill={CHART_PALETTE.green} />;
      })}
      {/* X labels — first + last yearMonth */}
      <Text x={padL} y={height - 4} style={{ fontSize: 7, fontFamily: "Montserrat", fill: CHART_PALETTE.brown }}>
        {series[0].yearMonth.slice(2).replace("-", "/")}
      </Text>
      <Text x={width - padR} y={height - 4} style={{ fontSize: 7, fontFamily: "Montserrat", fill: CHART_PALETTE.brown, textAnchor: "end" }}>
        {series[series.length - 1].yearMonth.slice(2).replace("-", "/")}
      </Text>
    </Svg>
  );
}
 
// ── Premium Curve — the loyalty curve with the customer's marker ────────
// Plots the premium % vs grams accumulated (1g–100g) and highlights where
// the customer sits today. Editorial visual showing how their next 1g
// rate fits in the bigger arc.
export function PremiumCurveVisualization({
  highlightGrams,
  curveFn,
  width  = 380,
  height = 160,
}: {
  highlightGrams: number;
  /** Returns premium % at given grams accumulated (cycle position). */
  curveFn: (grams: number) => number;
  width?: number;
  height?: number;
}) {
  const padL = 32, padR = 8, padT = 12, padB = 24;
  const innerW = width - padL - padR;
  const innerH = height - padT - padB;
  // Sample the curve from 1g → 100g
  const samples = Array.from({ length: 100 }, (_, i) => ({
    grams: i + 1,
    pct:   curveFn(i + 1),
  }));
  const maxPct = Math.max(...samples.map(s => s.pct));
  const minPct = Math.min(...samples.map(s => s.pct));
  const range = (maxPct - minPct) || 1;
  const xOf = (g: number) => padL + (g / 100) * innerW;
  const yOf = (p: number) => padT + (1 - (p - minPct) / range) * innerH;
 
  const linePts = samples.map(s => `${xOf(s.grams).toFixed(2)},${yOf(s.pct).toFixed(2)}`);
  const areaPts = [...linePts, `${xOf(100).toFixed(2)},${padT + innerH}`, `${xOf(0).toFixed(2)},${padT + innerH}`].join(" ");
 
  const cycleGrams = highlightGrams % 50;
  const customerPct = curveFn(Math.max(1, cycleGrams));
  const customerX   = xOf(cycleGrams);
  const customerY   = yOf(customerPct);
 
  return (
    <Svg width={width} height={height}>
      {/* Vertical milestone gridlines at 25g, 50g */}
      {[25, 50, 75, 100].map((g, i) => (
        <G key={i}>
          <Line x1={xOf(g)} y1={padT} x2={xOf(g)} y2={padT + innerH} stroke={CHART_PALETTE.subtle} strokeWidth={0.5} strokeDasharray="2,2" />
          <Text x={xOf(g)} y={height - 4} style={{ fontSize: 7, fontFamily: "Montserrat", fill: CHART_PALETTE.brown, textAnchor: "middle" }}>
            {g}g
          </Text>
        </G>
      ))}
      {/* Y-axis labels at min/max premium */}
      <Text x={padL - 4} y={padT + 4} style={{ fontSize: 7, fontFamily: "Montserrat", fill: CHART_PALETTE.brown, textAnchor: "end" }}>
        {maxPct.toFixed(1)}%
      </Text>
      <Text x={padL - 4} y={padT + innerH + 2} style={{ fontSize: 7, fontFamily: "Montserrat", fill: CHART_PALETTE.brown, textAnchor: "end" }}>
        {minPct.toFixed(1)}%
      </Text>
      {/* Area + curve */}
      <Polygon points={areaPts} fill={CHART_PALETTE.gold} fillOpacity={0.12} />
      <Polyline points={linePts.join(" ")} stroke={CHART_PALETTE.gold} strokeWidth={1.8} fill="none" />
      {/* Customer marker — vertical drop line + dot */}
      <Line x1={customerX} y1={customerY} x2={customerX} y2={padT + innerH} stroke={CHART_PALETTE.red} strokeWidth={1} strokeDasharray="2,2" />
      <Circle cx={customerX} cy={customerY} r={4} fill={CHART_PALETTE.red} />
      <Circle cx={customerX} cy={customerY} r={2} fill={CHART_PALETTE.paper} />
      {/* Customer label */}
      <Text x={customerX + 6} y={customerY - 4} style={{ fontSize: 7, fontFamily: "Montserrat", fontWeight: 700, fill: CHART_PALETTE.red }}>
        you · {customerPct.toFixed(2)}%
      </Text>
    </Svg>
  );
}
 
// ── Spend breakdown donut ───────────────────────────────────────────────
// Lifetime spend split across Premium Curve / Standard / Accessory.
// Centre shows the total. Arcs colored gold / blue / amber.
export function SpendDonut({
  segments,
  width = 140,
  height = 140,
}: {
  segments: { label: string; value: number; color: string }[];
  width?: number;
  height?: number;
}) {
  const total = segments.reduce((s, x) => s + x.value, 0);
  Iif (total <= 0) return null;
  const cx = width / 2;
  const cy = height / 2;
  const rOuter = Math.min(width, height) / 2 - 6;
  const rInner = rOuter * 0.62;
 
  let cumulative = 0;
  const arcs = segments
    .filter(s => s.value > 0)
    .map(s => {
      const startAngle = (cumulative / total) * 2 * Math.PI - Math.PI / 2;
      cumulative += s.value;
      const endAngle = (cumulative / total) * 2 * Math.PI - Math.PI / 2;
      return { ...s, startAngle, endAngle };
    });
 
  const arcPath = (start: number, end: number) => {
    const largeArc = end - start > Math.PI ? 1 : 0;

    const y1 = cy + rOuter * Math.sin(start);
    const x2 = cx + rOuter * Math.cos(end);
    const y2 = cy + rOuter * Math.sin(end);
    const x3 = cx + rInner * Math.cos(end);
    const y3 = cy + rInner * Math.sin(end);
    const x4 = cx + rInner * Math.cos(start);
    const y4 = cy + rInner * Math.sin(start);
    return [

      `A ${rOuter} ${rOuter} 0 ${largeArc} 1 ${x2.toFixed(2)} ${y2.toFixed(2)}`,
      `L ${x3.toFixed(2)} ${y3.toFixed(2)}`,
      `A ${rInner} ${rInner} 0 ${largeArc} 0 ${x4.toFixed(2)} ${y4.toFixed(2)}`,
      `Z`,
    ].join(" ");
  };
 
  return (
    <Svg width={width} height={height}>
      {arcs.map((a, i) => (
        <Path key={i} d={arcPath(a.startAngle, a.endAngle)} fill={a.color} />
      ))}
    </Svg>
  );
}
 
// ── Horizontal stacked bar ──────────────────────────────────────────────
// Used for "your share" comparisons or category breakdowns where a donut
// would be overkill. e.g. curve-vs-standard grams.
export function StackedBar({
  segments,
  width  = 380,
  height = 18,
  showLabels = true,
}: {
  segments: { label: string; value: number; color: string }[];
  width?: number;
  height?: number;
  showLabels?: boolean;
}) {
  const total = segments.reduce((s, x) => s + x.value, 0);
  if (total <= 0) return null;
  let x = 0;
  return (
    <Svg width={width} height={height + (showLabels ? 14 : 0)}>
      {segments.filter(s => s.value > 0).map((s, i) => {
        const w = (s.value / total) * width;
        const rect = <Rect key={i} x={x} y={0} width={w} height={height} fill={s.color} />;
        x += w;
        return rect;
      })}
      {showLabels && (() => {
        let lx = 0;
        return segments.filter(s => s.value > 0).map((s, i) => {
          const w = (s.value / total) * width;
          const cx = lx + w / 2;
          lx += w;
          return (
            <Text key={`l${i}`} x={cx} y={height + 10} style={{ fontSize: 7, fontFamily: "Montserrat", fill: CHART_PALETTE.brown, textAnchor: "middle" }}>
              {s.label} · {((s.value / total) * 100).toFixed(0)}%
            </Text>
          );
        });
      })()}
    </Svg>
  );
}
 
// ── Cycle progress bar ──────────────────────────────────────────────────
// Horizontal progress bar mirroring the app's GoldProgressBar — gold
// gradient fill against a subtle track, with a glow dot at the head.
// Replaces the earlier ring layout because the bar matches the hero
// progress bar customers already recognise from the dashboard.
export function CycleProgressBar({
  positionInCycle,
  width  = 360,
  height = 36,
  showLabels = true,
}: {
  positionInCycle: number;
  width?: number;
  height?: number;
  showLabels?: boolean;
}) {
  const target = 50; // 50g milestone cycle
  const pct = Math.min(1, Math.max(0, positionInCycle / target));
  const barH = 10;
  const barY = showLabels ? 14 : (height - barH) / 2;
  const trackR = barH / 2;
  const fillW = pct * width;
  const labelY = 8;
  const subY = barY + barH + 12;
  const dotR = 6;
  return (
    <Svg width={width} height={height}>
      {showLabels && (
        <>
          <Text x={0} y={labelY} style={{ fontSize: 7, fontFamily: "Montserrat", fontWeight: 700, fill: CHART_PALETTE.brown, letterSpacing: 1 }}>
            {positionInCycle.toFixed(positionInCycle % 1 === 0 ? 0 : 2)}g IN CYCLE
          </Text>
          <Text x={width} y={labelY} style={{ fontSize: 7, fontFamily: "Montserrat", fontWeight: 700, fill: CHART_PALETTE.brown, letterSpacing: 1, textAnchor: "end" }}>
            50g — NEXT MILESTONE
          </Text>
        </>
      )}
      {/* Track */}
      <Rect x={0} y={barY} width={width} height={barH} rx={trackR} ry={trackR} fill={CHART_PALETTE.subtle} />
      {/* Fill */}
      {fillW > 0 && (
        <Rect x={0} y={barY} width={fillW} height={barH} rx={trackR} ry={trackR} fill={CHART_PALETTE.gold} />
      )}
      {/* Glow dot at the head */}
      {fillW > 0 && (
        <>
          <Circle cx={fillW} cy={barY + barH / 2} r={dotR} fill={CHART_PALETTE.gold} />
          <Circle cx={fillW} cy={barY + barH / 2} r={dotR - 2} fill={CHART_PALETTE.paper} />
        </>
      )}
      {showLabels && (
        <Text x={fillW} y={subY} style={{ fontSize: 7, fontFamily: "Montserrat", fill: CHART_PALETTE.brown, textAnchor: fillW < 60 ? "start" : fillW > width - 60 ? "end" : "middle" }}>
          {(target - positionInCycle).toFixed(2)}g to go
        </Text>
      )}
    </Svg>
  );
}
 
// ── Legacy cycle progress ring (kept for compatibility, not used) ───────
export function CycleProgressRing({
  positionInCycle,
  width  = 90,
  height = 90,
}: {
  positionInCycle: number;
  width?: number;
  height?: number;
}) {
  const cx = width / 2;
  const cy = height / 2;
  const r  = Math.min(width, height) / 2 - 8;
  const pct = Math.min(1, Math.max(0, positionInCycle / 50));
  const startAngle = -Math.PI / 2;
  const endAngle   = startAngle + pct * 2 * Math.PI;
 
  // Background ring
  // Foreground arc
  const x1 = cx + r * Math.cos(startAngle);
  const y1 = cy + r * Math.sin(startAngle);
  const x2 = cx + r * Math.cos(endAngle);
  const y2 = cy + r * Math.sin(endAngle);
  const largeArc = pct > 0.5 ? 1 : 0;
  const arc = `M ${x1.toFixed(2)} ${y1.toFixed(2)} A ${r} ${r} 0 ${largeArc} 1 ${x2.toFixed(2)} ${y2.toFixed(2)}`;
 
  return (
    <Svg width={width} height={height}>
      <Circle cx={cx} cy={cy} r={r} fill="none" stroke={CHART_PALETTE.subtle} strokeWidth={5} />
      <Path d={arc} fill="none" stroke={CHART_PALETTE.gold} strokeWidth={5} strokeLinecap="round" />
      <Text x={cx} y={cy + 1} style={{ fontSize: 13, fontFamily: "Montserrat", fontWeight: 700, fill: CHART_PALETTE.brown, textAnchor: "middle" }}>
        {positionInCycle.toFixed(positionInCycle % 1 === 0 ? 0 : 1)}g
      </Text>
      <Text x={cx} y={cy + 12} style={{ fontSize: 7, fontFamily: "Montserrat", fill: CHART_PALETTE.brown, textAnchor: "middle" }}>
        of 50g cycle
      </Text>
    </Svg>
  );
}
 
// ── Premium Curve closed-form for the visualization above ──────────────
// Mirrors src/lib/premiums.ts logic but kept self-contained so the chart
// helper doesn't pull a network of imports. Keep in sync if curve shape
// changes there.
export function defaultCurveFn(grams: number): number {
  // Approximate match to the production premium curve: starts ~5.5% at
  // 1g and decays to ~0.5% at 50g, then flat.
  const cap = Math.min(grams, 50);
  const top = 5.5;
  const bot = 0.5;
  // exponential decay
  const k = -Math.log(0.05) / 50;
  const decayed = bot + (top - bot) * Math.exp(-k * cap);
  return decayed;
}
