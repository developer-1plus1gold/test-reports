// ── Script-aware hyphenation / line-breaking ───────────────────────────
// @react-pdf's default word-break treats every space-delimited token as
// indivisible. That works for Latin (with `-` opportunities) but breaks
// badly for non-Latin scripts:
//   • CJK has no spaces → a paragraph reads as one unbreakable "word"
//     and overflows the right margin (Mukesh saw this on Mandarin PDFs).
//   • Devanagari/Bengali/Tamil HAVE spaces, but conjunct clusters
//     (e.g. Devanagari "क्ष", Bengali "ক্ষ") are multi-codepoint
//     graphemes that must NEVER split mid-cluster. A naive "break
//     anywhere" rule shears conjuncts across two lines.
//   • And just keeping Indic words intact creates its own bug: a long
//     compound word can exceed the column width (Mukesh saw this on
//     Bengali — text bled past the right padding even with hyphenation
//     suppressed) because the engine has nowhere to break.
//
// Fix: split Indic words into Unicode GRAPHEMES via Intl.Segmenter.
// Graphemes correctly group conjuncts as single units, so the engine
// can break between any two graphemes without ever cutting inside
// one. CJK uses the same trick (per-character) since each character
// is its own grapheme. Latin keeps default behaviour.
//
// One callback covers every text element in the document.
const indicSegmenter = typeof Intl !== "undefined" && "Segmenter" in Intl
  ? new Intl.Segmenter(undefined, { granularity: "grapheme" })
  : null;

/** Returns the font family to use for body prose + data labels in the
 *  given language. Brand fonts cover Latin scripts; non-Latin scripts
 *  fall back to the Noto family above. The Cinzel masthead stays
 *  unchanged — "1PLUS1GOLD" is a wordmark, no need to translate. */
function bodyFontFamily(lang: LanguageCode | undefined): string {
  switch (lang) {
    case "zh-Hans": return "Noto Sans SC";
    case "bn":      return "Noto Sans Bengali";
    case "ta":      return "Noto Sans Tamil";
    case "hi":      return "Noto Sans Devanagari";
    default:        return "Cormorant"; // en, ms, tl, id — Latin script
  }
}

// Build the StyleSheet at render time so fontFamily switches based on
// the viewer's language. Latin scripts (en, ms, tl, id) get the brand
// fonts (Cinzel + Cormorant + Montserrat). Non-Latin scripts (zh-Hans,

// script — without this swap, child <Text> elements with explicit
// fontFamily values rendered as boxes for non-Latin viewers because
// Cinzel/Cormorant/Montserrat don't carry CJK / Bengali / Tamil /
// Devanagari glyphs. The "1PLUS1GOLD" wordmark stays Cinzel always.
function buildStyles(language: LanguageCode | undefined) {

  // language-specific Noto family. For Latin: keep brand fonts.
  // Display headings (Cinzel) also swap to Noto for non-Latin so
  // translated section eyebrows / titles render properly.
  const noto = lat ? null : bodyFontFamily(language);

  const fBody    = noto ?? "Cormorant"; // editorial prose
  const fData    = noto ?? "Montserrat"; // labels, figures

// ── Format helpers ────────────────────────────────────────────────────────
// Two flavors of currency formatting:
//  - fmtSgd(n) :: legacy helper, kept only for the few internal call
//                 sites where we deliberately want SGD without
//                 conversion (e.g. footnote disclosures).
//  - mkFmt(currency, rates) :: closure factory used inside the Document
//                 render. Converts SGD source values to display currency
//                 + applies the right symbol/decimals from CURRENCIES.
function fmtSgd(n: number, currency = "SGD"): string {
  const sym = currency === "SGD" ? "S$" : currency === "USD" ? "US$" : `${currency} `;
  return sym + " " + n.toFixed(2).replace(/\B(?=(\d{3})+(?!\d))/g, ",");
}
/** Build a "format an SGD amount in the viewer's currency" closure. */
function makeMoneyFmt(displayCurrency: string, rates: Record<string, number>) {
  const cfg = CURRENCIES[displayCurrency] ?? CURRENCIES.SGD;
  const rate = rates[displayCurrency] ?? 1;
  return (sgdAmount: number) => {
    const converted = sgdAmount * rate;
    const formatted = converted.toFixed(cfg.decimals).replace(/\B(?=(\d{3})+(?!\d))/g, ",");
    return `${cfg.symbol} ${formatted}`;
  };
}
function fmtG(n: number, dp?: number): string {
  const d = dp ?? (n % 1 === 0 ? 0 : 2);
  return `${n.toFixed(d)}g`;
}
/** Bare grams number (no `g` suffix) — drops trailing ".00" on whole
 *  values so labels like "{g}g to next conversion" don't render as
 *  "50.00g to next conversion". Used inside fmtLabel substitutions
 *  where the label already carries the `g` suffix. */
function fmtGBare(n: number): string {
  return n % 1 === 0 ? `${n.toFixed(0)}` : `${n.toFixed(2)}`;
}
function fmtPct(pct: number, dp = 2): string { return `${pct.toFixed(dp)}%`; }
function fmtSignedPct(pct: number): string {
  return `${pct > 0 ? "+" : ""}${pct.toFixed(2)}%`;
}
function fmtDateTimeSGT(iso: string): string {
  return new Date(iso).toLocaleString("en-SG", {
    day: "2-digit", month: "short", year: "numeric",
    hour: "2-digit", minute: "2-digit", hour12: true,
    timeZone: "Asia/Singapore",
  }) + " SGT";
}
function mdToParas(md: string): string[] {
  // Strip markdown noise + split into paragraphs for editorial body.
  return md
    .replace(/^#{1,6}\s+/gm, "")
    .replace(/\*\*(.+?)\*\*/g, "$1")
    .replace(/\*(.+?)\*/g, "$1")
    .replace(/`(.+?)`/g, "$1")
    .replace(/\[(.+?)\]\(.+?\)/g, "$1")
    .replace(/^[-*]\s+/gm, "• ")
    .trim()
    .split(/\n{2,}/)
    .map(p => p.trim())
    .filter(Boolean);
}

                <Text style={[styles.brickTitle, { color: blue, marginBottom: 5 }]}>{L.standardBarsTitle}</Text>
                <Text style={styles.proseSmall}>
                  {fmtLabel(L.standardBarsBodyFmt, {
                    grams: fmtG(c.standard.gramsLifetime),
                    avgPremiumNote: c.standard.avgPremiumPct != null
                      ? fmtLabel(L.standardBarsAvgPremiumFmt, { pct: fmtPct(c.standard.avgPremiumPct) })
                      : "",
                  })}
                </Text>
              </View>
            )}
 
            {hasAccessory && (
              <View wrap={false} style={{ marginBottom: 14 }}>
                <Text style={[styles.brickTitle, { color: amber, marginBottom: 5 }]}>{L.accessoriesTitle}</Text>
                <Text style={styles.proseSmall}>
                  {fmtLabel(L.accessoriesBodyFmt, {
                    n: c.accessory.itemsLifetime,
                    totalSpend: money(c.accessory.lifetimeSpendSgd),
                    thisMonthNote: c.accessory.spendThisMonth > 0
                      ? fmtLabel(L.accessoriesThisMonthFmt, { monthSpend: money(c.accessory.spendThisMonth) })
                      : "",
                  })}
                </Text>
              </View>
            )}
 
            {hasGoldYield && c.goldYield && (
              <View wrap={false} style={{ marginBottom: 14 }}>
                <Text style={[styles.brickTitle, { color: green, marginBottom: 5 }]}>{L.goldYieldTitle}</Text>
                <Text style={styles.proseSmall}>
                  {fmtLabel(L.goldYieldBodyFmt, {
                    bars: c.goldYield.barsEnrolled,
                    grams: fmtG(c.goldYield.gramsYielded),
                    earningsNote: c.goldYield.earningsPaidSgd > 0
                      ? fmtLabel(L.goldYieldEarningsPaidFmt, { paid: money(c.goldYield.earningsPaidSgd) })
                      : L.goldYieldEarningsPending,
                  })}
                </Text>
              </View>
            )}
 
            {hasBuyback && c.buyback && (
              <View wrap={false} style={{ marginBottom: 14 }}>
                <Text style={[styles.brickTitle, { color: red, marginBottom: 5 }]}>{L.buybackTitle}</Text>
                <Text style={styles.proseSmall}>
                  {fmtLabel(L.buybackBodyFmt, {
                    grams: fmtG(c.buyback.totalGramsSold),
                    payoutNote: c.buyback.totalPayoutSgd > 0
                      ? fmtLabel(L.buybackPayoutFmt, { payout: money(c.buyback.totalPayoutSgd) })
                      : L.buybackPayoutPending,
                    thisMonthNote: c.buyback.gramsSoldThisMonth > 0
                      ? fmtLabel(L.buybackThisMonthFmt, { grams: fmtG(c.buyback.gramsSoldThisMonth) })
                      : "",
                  })}
                </Text>
              </View>
            )}
 
            {hasConversion && c.conversion && (
              <View wrap={false} style={{ marginBottom: 14 }}>
                <Text style={[styles.brickTitle, { color: gold, marginBottom: 5 }]}>{L.conversionsTitle}</Text>
                <Text style={styles.proseSmall}>
                  {fmtLabel(L.conversionsBodyFmt, {
                    n: c.conversion.count50g + c.conversion.count100g,
                    breakdown: `${c.conversion.count50g}× 50g${c.conversion.count100g > 0 ? `, ${c.conversion.count100g}× 100g` : ""}`,
                    feesNote: c.conversion.totalFeesSgd > 0
                      ? fmtLabel(L.conversionsFeesFmt, { fees: money(c.conversion.totalFeesSgd) })
                      : "",
                  })}
                </Text>
              </View>
            )}
          </>
        )}
 
        <FooterBar note={`Your gold story · ${periodLabel}`} />
      </Page>
 
      {/* ════════════════════════ PAGE 4 — THE MARKET ═══════════════════ */}
      <Page size="A4" style={pageStyle} wrap>
        <Text style={styles.sectionEyebrow}>{periodLabel}</Text>
        <Text style={styles.sectionTitle}>{L.marketSectionTitle}</Text>
        <View style={styles.sectionRule} />
 
        {/* Headline */}
        <Text style={styles.prose}>{ms.headline}</Text>
 
        {/* Spot trend chart — full width */}
        {hasSpotSeries && (
          <View wrap={false} style={{ marginTop: 8, marginBottom: 14 }}>
            <Text style={styles.subsectionHead}>{fmtLabel(L.marketSpotTrendSubheadFmt, { month: monthName, year: data.brief.period.year })}</Text>
            <LineChart
              data={c.spotSeries.map(p => p.sgdPerG)}
              width={460}
              height={140}
              color={upColor}
              yLabelFmt={(v) => `S$${v.toFixed(0)}`}
              xLabelFirst={`${monthName.slice(0, 3)} 1`}
              xLabelLast={`${monthName.slice(0, 3)} ${c.spotSeries.length}`}
            />
            <Text style={styles.caption}>
              {fmtLabel(L.dailySpotCloseCaptionFmt, {
                pctNote: hasSpotMove && c.spotMove.pctChange != null
                  ? fmtLabel(L.dailySpotClosePctFmt, { pct: fmtSignedPct(c.spotMove.pctChange) })
                  : "",
              })}
            </Text>
          </View>
        )}
 
        {/* Numbers */}
        {ms.spot_open_usd_oz != null && ms.spot_close_usd_oz != null && (
          <View wrap={false}>
            <Text style={styles.subsectionHead}>{fmtLabel(L.marketNumbersSubheadFmt, { month: monthName, year: data.brief.period.year })}</Text>
            <StatRow>
              <Stat
                label={fmtLabel(L.statMonthOpenFmt, { monthShort: monthName.slice(0, 3) })}
                value={pricePerG(ms.spot_open_usd_oz)}
                sub={fmtLabel(L.statMonthOpenSubFmt, { usd: ms.spot_open_usd_oz.toFixed(2) })}
              />
              <Stat
                label={fmtLabel(L.statMonthCloseFmt, { monthShort: monthName.slice(0, 3) })}
                value={pricePerG(ms.spot_close_usd_oz)}
                sub={fmtLabel(L.statMonthCloseSubFmt, { usd: ms.spot_close_usd_oz.toFixed(2) })}
              />
              {ms.spot_high_usd_oz != null && (
                <Stat
                  label={fmtLabel(L.statMonthHighFmt, { monthShort: monthName.slice(0, 3) })}
                  value={pricePerG(ms.spot_high_usd_oz)}
                  sub={fmtLabel(L.statMonthHighSubFmt, { usd: ms.spot_high_usd_oz.toFixed(2) })}
                />
              )}
              {ms.pct_change != null && (
                <Stat
                  label={L.statMonthChange}
                  value={fmtSignedPct(ms.pct_change)}
                  sub={L.statMonthChangeSub}
                  valueColor={ms.pct_change >= 0 ? green : red}
                />
              )}
            </StatRow>
            <Text style={styles.caption}>
              {displayCurrency === "SGD"
                ? fmtLabel(L.fxDisclosureSGDFmt, { month: monthName, fxRate: fxSgdUsd.toFixed(4) })
                : fmtLabel(L.fxDisclosureOtherFmt, { currency: displayCurrency, month: monthName, fxRate: fxSgdUsd.toFixed(4) })}
            </Text>
          </View>
        )}
 
        {/* Three drivers */}
        <View wrap={false} style={{ marginTop: 16 }}>
          <Text style={styles.subsectionHead}>{L.whatDroveItSubhead}</Text>
          {data.brief.three_drivers.map((d, i) => {
            const accent = [gold, blue, green][i % 3];
            return <Brick key={i} title={d.title} body={d.explanation} accent={accent} />;
          })}
        </View>
 
        {/* Did you know */}
        <View wrap={false} style={{ marginTop: 8 }}>
          <Text style={styles.subsectionHead}>{fmtLabel(L.didYouKnowSubheadFmt, { term: data.brief.concept_explainer.term })}</Text>
          <Text style={styles.prose}>{data.brief.concept_explainer.explanation}</Text>
        </View>
 
        {/* Things worth knowing */}
        {data.brief.things_worth_knowing.length > 0 && (
          <View wrap={false} style={{ marginTop: 8 }}>
            <Text style={styles.subsectionHead}>{L.thingsWorthKnowingSubhead}</Text>
            {data.brief.things_worth_knowing.map((t, i) => {
              const accent = [amber, blue, gold][i % 3];
              return <Brick key={i} title={t.topic} body={t.explanation} accent={accent} />;
            })}
          </View>
        )}
 
        {/* Looking ahead — closing pace + soft call-to-action */}
        <View wrap={false} style={{ marginTop: 16, paddingTop: 14, borderTopWidth: 0.5, borderTopColor: rule }}>
          <Text style={styles.subsectionHead}>{fmtLabel(L.whatsNextSubheadFmt, { name: c.firstName })}</Text>
          {(c.monthsToNext50g != null || c.monthsToNext100g != null || c.curve.splitsInDrawerGrams >= 50) && (
            <StatRow>
              {c.monthsToNext50g != null && (
                <Stat label={L.statToNext50g} value={`${c.monthsToNext50g} mo`} sub={L.statToNextSub} valueColor={c.monthsToNext50g <= 3 ? green : c.monthsToNext50g <= 6 ? amber : ink} />
              )}
              {c.monthsToNext100g != null && (
                <Stat label={L.statToNext100g} value={`${c.monthsToNext100g} mo`} sub={L.statToNextSub} />
              )}
              {c.curve.splitsInDrawerGrams >= 50 && (
                <Stat label={L.statReadyNow} value={`${Math.floor(c.curve.splitsInDrawerGrams / 50)}× 50g`} sub={L.statReadyNowSub} valueColor={amber} />
              )}
              <Stat label={L.statNext1gRate} value={c.curve.nextPurchaseRatePct != null ? fmtPct(c.curve.nextPurchaseRatePct) : "—"} sub={L.statNext1gRateSub} valueColor={amber} />
            </StatRow>
          )}
          {/* Primary nudge — what's the most actionable thing for this
              specific customer right now? Conversion-ready beats pace
              beats general curve nudge. Phrased as the editor's
              recommendation, not a hard sell. */}
          <Text style={styles.prose}>
            {c.curve.splitsInDrawerGrams >= 50
              ? fmtLabel(L.nudgeReadyFmt, {
                  grams:       fmtG(c.curve.splitsInDrawerGrams),
                  conversions: Math.floor(c.curve.splitsInDrawerGrams / 50),
                })
              : c.monthsToNext50g != null && c.monthsToNext50g > 0 && c.monthsToNext50g <= 3
                ? fmtLabel(L.nudgeNearFmt, { months: c.monthsToNext50g })
                : c.monthsToNext50g != null && c.monthsToNext50g > 0
                  ? fmtLabel(L.nudgeMidFmt, { months: c.monthsToNext50g })
                  : L.nudgeFar}
          </Text>
          <Text style={[styles.prose, { fontSize: 9.5, color: inkSoft, marginTop: 6 }]}>
            {L.closingKickerNote}
          </Text>
          <Text style={[styles.proseMuted, { marginTop: 14, textAlign: "center", fontSize: 11 }]}>
            {L.closingTagline}
          </Text>
        </View>
 
        {/* Compliance line at the very foot of last page */}
        <Text style={[styles.caption, { marginTop: 24, textAlign: "center" }]}>
          {L.footerCompliance}
        </Text>
 
        <FooterBar note={`The market · ${periodLabel}`} />
      </Page>
 
    </Document>
  );
