/**
 * Customer-section block library.
 *
 * Each block has:
 *   - `kind` — stable identifier for analytics and admin overrides
 *   - `condition` — predicate over CustomerNewsletterData
 *   - `template` — string with `{placeholder}` substitutions
 *
 * Block evaluation is deterministic. No LLM call. The block library is the
 * single source of truth for what customers see in the "Your month" section.
 *
 * Adding/removing blocks is a code change, version-controlled, reviewed.
 *
 * Blocks fire in declaration order — first matching block per slot fires.
 * Multiple blocks can fire in different slots. Slot ordering produces final
 * display order.
 */
 
import { EN_LABELS, fmt as fmtLabel } from "./labels";
 
export interface CustomerNewsletterData {
  customerId:          string;
  firstName:           string;
  /** Full name from the profile. Used on the masthead recipient line
   *  (PDF + page) where the full name reads better than firstName for
   *  formal correspondence. Empty string when profile has no full_name. */
  fullName:            string;
  /** Period this newsletter covers (the prior month, since cron runs on 1st). */
  periodYear:          number;
  periodMonth:         number;
  /** Customer's tenure on the platform, in calendar months. */
  tenureMonths:        number;
  /** Total grams accumulated as of newsletter view time.
   *  HISTORICAL FIELD — kept for the existing block library.
   *  Equals curve + standard lifetime grams (mixed). New PDF surfaces should
   *  prefer the typed fields below. */
  totalGrams:          number;
  /** Grams added during the period (curve + standard combined). */
  gramsAddedThisMonth: number;
  /** Number of distinct purchases in the period. */
  purchasesThisMonth:  number;
  /** Average premium % across the customer's purchases this calendar year. */
  avgPremiumPctYTD:    number | null;
  /** Months of consecutive purchase activity ending in the period. */
  purchaseStreakMonths: number;
  /** Months since the customer's most recent purchase (0 = bought this month). */
  monthsSinceLastPurchase: number;
  /** True if customer crossed any milestone this month. */
  milestone: 'first_10g' | 'halfway_to_50g' | 'first_combibar_50g' | 'first_combibar_100g' | 'first_purchase' | null;
  /** Tree branches the customer has, with allocated grams + goal grams. */
  treeBranches: { name: string; allocatedGrams: number; goalGrams: number; firedThisMonth: boolean }[];
  /** Pace projection — months to next 50g / 100g milestone at trailing 3-month avg. */
  monthsToNext50g:  number | null;
  monthsToNext100g: number | null;
  /** True if customer recently returned after 3+ months of no activity. */
  returnedAfterGap: boolean;
  /** True if customer joined this calendar month. */
  joinedThisMonth:  boolean;
  /** True if customer is dormant (no purchase + no login >3 months) but still has grams. */
  dormantWithBalance: boolean;
 
  // ── Per-section data for the PDF newsletter ────────────────────────────
  // The fields below were added 2026-05-05 so the newsletter PDF can show
  // the customer's full picture (matches what they see on the dashboard
  // and order history). The block library above doesn't need them — only
  // the PDF generator reads them — but they're declared here to keep the
  // type as a single source of truth.
 
  /** Curve products journey — drives 50g milestones, premium curve, splits drawer. */
  curve: {
    /** Lifetime curve-product grams (curve_grams_purchased). Loyalty-anchored. */
    gramsLifetime:        number;
    /** Cycle position = gramsLifetime % 50. */
    positionInCycle:      number;
    /** floor(gramsLifetime / 50). */
    cyclesCompleted:      number;
    /** Lifetime SGD spent on curve items (price_sgd × qty across all curve lines). */
    lifetimeSpendSgd:     number;
    /** lifetimeSpendSgd / gramsLifetime. NULL if no curve purchases. */
    avgCostPerGramSgd:    number | null;
    /** Grams-weighted average premium across curve items, sane-banded. */
    avgPremiumPct:        number | null;
    /** Rate the next 1g curve purchase will land on (curve(min(pos+1, 50))). */
    nextPurchaseRatePct:  number | null;
    /** Grams sitting as 1g/5g/10g splits (lifetime curve − physical 50/100 grams). */
    splitsInDrawerGrams:  number;
    /** Curve grams added during the period. */
    gramsAddedThisMonth:  number;
  };
 
  /** Standard products journey — independent qty-tier pricing. */
  standard: {
    gramsLifetime:        number;
    itemsLifetime:        number;
    lifetimeSpendSgd:     number;
    avgCostPerGramSgd:    number | null;
    avgPremiumPct:        number | null;
    gramsAddedThisMonth:  number;
  };
 
  /** Accessories — items + spend, no premium concept. */
  accessory: {
    itemsLifetime:        number;
    lifetimeSpendSgd:     number;
    spendThisMonth:       number;
  };
 
  /** Gold Yield programme activity. NULL if customer is not enrolled / no bars. */
  goldYield: {
    barsEnrolled:         number;
    gramsYielded:         number;
    earningsPaidSgd:      number;
  } | null;
 
  /** Buyback (sell-to-us) activity. NULL if customer has never sold back. */
  buyback: {
    totalGramsSold:       number;
    totalPayoutSgd:       number;
    avgPerGramSgd:        number | null;
    gramsSoldThisMonth:   number;
  } | null;
 
  /** Combibar conversions (1g splits → sealed bars). NULL if none completed. */
  conversion: {
    count50g:             number;
    count100g:            number;
    totalFeesSgd:         number;
    completedThisMonth:   number;
  } | null;
 
  /** Spot price for the period in the customer's preferred currency.
   *  Used by the page-1 hero "Gold +X% this month" tile. NULL if history
   *  doesn't have rows covering both ends of the period. */
  spotMove: {
    currency:    string;       // ISO code, e.g. "SGD" or customer's preferred
    openPerG:    number | null;
    closePerG:   number | null;
    pctChange:   number | null;
  };
 
  /** Daily spot prices (SGD/g) for each trading day in the period. Feeds
   *  the spot-trend line chart on the market page. May be empty if
   *  gold_price_history doesn't have rows for this window. */
  spotSeries: { date: string; sgdPerG: number }[];
 
  /** Trailing 12-month accumulation buckets ending at the period.
   *  Each entry has {yearMonth, gramsAdded, gramsCumulative}. Drives the
   *  accumulation line/area chart on the customer's data page. */
  monthlyAccumulation: { yearMonth: string; gramsAdded: number; gramsCumulative: number }[];
 
  /** Customer's preferred currency for SGD-equivalent display. Defaults SGD. */
  preferredCurrency: string;
}
 
export type BlockKind =
  | 'welcome_first_purchase'
  | 'welcome_no_purchase'
  | 'returning_after_gap'
  | 'milestone_first_purchase'
  | 'milestone_first_10g'
  | 'milestone_halfway_to_50g'
  | 'milestone_combibar_50g'
  | 'milestone_combibar_100g'
  | 'purchased_this_month'
  | 'streak'
  | 'steady_no_purchase'
  | 'dormant_with_balance'
  | 'premium_savings'
  | 'tree_allocated_this_month'
  | 'pace_to_50g'
  | 'pace_to_100g';
 
export interface RenderedBlock {
  kind: BlockKind;
  /** EN_LABELS key the renderer reads (via mergeLabels for the viewer's
   *  language) and substitutes vars into. */
  labelKey: keyof import("./labels").NewsletterLabels;
  /** Substitution variables for the label template. */
  vars: Record<string, string | number>;
  /** Pre-formatted English text (sample/compliance path uses this). The
   *  renderer prefers labelKey + vars so non-English readers get the
   *  translated template. */
  text: string;
}
 
/**
 * Evaluate the block library against customer data; return the blocks that
 * fire, in display order.
 *
 * Each block emits {kind, labelKey, vars}. The renderer reads
 * L[labelKey] (using mergeLabels with the viewer's language) and
 * substitutes vars at render time, so non-English readers see the
 * translated text. `text` is pre-substituted English for the
 * sample/compliance path that doesn't have a viewer language.
 */
export function renderBlocks(d: CustomerNewsletterData): RenderedBlock[] {
  const out: RenderedBlock[] = [];
  const fmtG = (g: number) => `${g.toFixed(g % 1 === 0 ? 0 : 2)}g`;
  const push = (kind: BlockKind, labelKey: keyof import("./labels").NewsletterLabels, vars: Record<string, string | number>) => {
    const text = fmtLabel(EN_LABELS[labelKey], vars);
    out.push({ kind, labelKey, vars, text });
  };
 
  // 1) Welcome / first-month
  if (d.joinedThisMonth) {
    if (d.purchasesThisMonth > 0) {
      push('welcome_first_purchase', 'blockWelcomeFirstPurchaseFmt', {
        name: d.firstName,
        grams: fmtG(d.gramsAddedThisMonth),
      });
    } else {
      push('welcome_no_purchase', 'blockWelcomeNoPurchaseFmt', { name: d.firstName });
    }
  } else if (d.returnedAfterGap) {
    push('returning_after_gap', 'blockReturningAfterGapFmt', {
      name: d.firstName,
      grams: fmtG(d.totalGrams),
    });
  }
 
  // 2) Milestones — strongest signal, fire prominently
  switch (d.milestone) {
    case 'first_purchase':
      // Already covered by welcome_first_purchase; skip duplicate.
      break;
    case 'first_10g':
      push('milestone_first_10g', 'blockMilestoneFirst10g', {});
      break;
    case 'halfway_to_50g': {
      // Compute the pace sub-clause via fmtLabel so non-English viewers
      // get a fully-translated sentence: outer template references
      // {pace}, the inner is its own translated label.
      const pace = d.monthsToNext50g != null
        ? fmtLabel(EN_LABELS.blockMilestoneHalfway50gPaceFmt, { months: d.monthsToNext50g })
        : EN_LABELS.blockMilestoneHalfway50gNoPace;
      push('milestone_halfway_to_50g', 'blockMilestoneHalfway50gFmt', { pace });
      break;
    }
    case 'first_combibar_50g':
      push('milestone_combibar_50g', 'blockMilestoneCombibar50g', {});
      break;
    case 'first_combibar_100g':
      push('milestone_combibar_100g', 'blockMilestoneCombibar100gFmt', { grams: fmtG(100) });
      break;
  }
 
  // 3) Activity / accumulation
  if (!d.joinedThisMonth) {
    if (d.purchasesThisMonth > 0) {
      push('purchased_this_month', 'blockPurchasedThisMonthFmt', {
        grams: fmtG(d.gramsAddedThisMonth),
        n:     d.purchasesThisMonth,
        total: fmtG(d.totalGrams),
      });
    } else if (d.dormantWithBalance) {
      push('dormant_with_balance', 'blockDormantWithBalanceFmt', { grams: fmtG(d.totalGrams) });
    } else if (Ed.totalGrams > 0) {
      push('steady_no_purchase', 'blockSteadyNoPurchaseFmt', { grams: fmtG(d.totalGrams) });
    }
  }
 
  // Streak callout (independent of activity above)
  if (d.purchaseStreakMonths >= 3) {
    push('streak', 'blockStreakFmt', { months: d.purchaseStreakMonths });
  }
 
  // 4) Premium savings — only if meaningfully below market median
  if (d.avgPremiumPctYTD != null && d.avgPremiumPctYTD < 4 && d.totalGrams > 0) {
    push('premium_savings', 'blockPremiumSavingsFmt', {
      pct: `${d.avgPremiumPctYTD.toFixed(2)}%`,
    });
  }
 
  // 5) Tree progress — only branches that received gold this month
  for (const b of d.treeBranches) {
    if (!b.firedThisMonth) continue;
    const pct = b.goalGrams > 0 ? Math.round((b.allocatedGrams / b.goalGrams) * 100) : 0;
    if (pct > 0 && pct < 100) {
      push('tree_allocated_this_month', 'blockTreeProgressFmt', {
        name: b.name,
        pct:  `${pct}%`,
        goal: fmtG(b.goalGrams),
      });
    } else if (Epct >= 100) {
      push('tree_allocated_this_month', 'blockTreeReachedFmt', {
        name: b.name,
        goal: fmtG(b.goalGrams),
      });
    } else {
      push('tree_allocated_this_month', 'blockTreeUngoaledFmt', { name: b.name });
    }
  }
 
  // 6) Pace projections — only if customer has any accumulation and a recent purchase
  if (d.totalGrams >= 30 && d.totalGrams < 50 && d.monthsToNext50g != null && d.monthsToNext50g > 0) {
    push('pace_to_50g', 'blockPaceTo50gFmt', { months: d.monthsToNext50g });
  } else if (d.totalGrams >= 50 && d.totalGrams < 100 && d.monthsToNext100g != null && d.monthsToNext100g > 0) {
    push('pace_to_100g', 'blockPaceTo100gFmt', { months: d.monthsToNext100g });
  }
 
  return out;
}
