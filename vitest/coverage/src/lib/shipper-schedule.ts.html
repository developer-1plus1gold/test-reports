export type ScheduleStatus = {

};
 
/**
 * Compute SGT day-of-week (0=Sun..6=Sat) and minutes since midnight for the
 * given Date. Uses Asia/Singapore.
 */
export function nowInSgt(now: Date = new Date()): { dow: number; minutes: number } {
  // en-GB to get DD/MM/YYYY without locale ambiguity, weekday short for day name.
  const parts = new Intl.DateTimeFormat("en-GB", {
    timeZone: "Asia/Singapore",
    weekday:  "short",
    hour:     "2-digit",
    minute:   "2-digit",
    hour12:   false,
  }).formatToParts(now);
  const wd = parts.find((p) => p.type === "weekday")?.value ?? "Sun";
  const hh = parseInt(parts.find((p) => p.type === "hour")?.value ?? "0", 10);
  const mm = parseInt(parts.find((p) => p.type === "minute")?.value ?? "0", 10);
  const dowMap: Record<string, number> = { Sun: 0, Mon: 1, Tue: 2, Wed: 3, Thu: 4, Fri: 5, Sat: 6 };
  return { dow: dowMap[wd] ?? 0, minutes: hh * 60 + mm };
}

function trimSeconds(t: string | null | undefined): string | null {
  if (!t) return null;
  const m = /^(\d{2}:\d{2})/.exec(t);
  return m ? m[1] : null;
}
 
export function evaluateScheduleStatus(
  row: ShipperScheduleRow | null | undefined,
  now: Date = new Date(),
): ScheduleStatus {
  if (!row) {
    // No schedule row → default open (fail-safe; defaults are seeded but new
    // shippers added later won't break customer flow if seed is forgotten).
    return { available: true, reason: null, postCutoff: false, todayOpen: null, todayClose: null, todayCutoff: null };
  }
 
  const todayOpen   = trimSeconds(row.open_time);
  const todayClose  = trimSeconds(row.close_time);
  const todayCutoff = trimSeconds(row.next_day_cutoff);
 
  if (!row.enabled) {
    return { available: false, reason: "closed_day", postCutoff: false, todayOpen, todayClose, todayCutoff };
  }
  const { minutes } = nowInSgt(now);
  const open   = timeToMinutes(row.open_time)  ?? 0;
  const close  = timeToMinutes(row.close_time) ?? 24 * 60;
  if (minutes < open)  return { available: false, reason: "before_open", postCutoff: false, todayOpen, todayClose, todayCutoff };
  if (minutes >= close) return { available: false, reason: "after_close", postCutoff: false, todayOpen, todayClose, todayCutoff };
 
  const cutoff = timeToMinutes(row.next_day_cutoff);
  const postCutoff = cutoff !== null && minutes >= cutoff;
  return { available: true, reason: "open", postCutoff, todayOpen, todayClose, todayCutoff };
}
 
/**
 * Fetch today's schedule for the named shippers and return the per-shipper
 * evaluation. Quiet on missing rows — returns "available" defaults so a
 * forgotten seed never blocks checkout.
 */
export async function fetchTodayShipperStatuses(
  shippers: string[],
  now: Date = new Date(),
): Promise<Record<string, ScheduleStatus>> {
  const { dow } = nowInSgt(now);
  const { data, error } = await getAdminClient()
    .from("shipper_schedules")
    .select("shipper, day_of_week, enabled, open_time, close_time, next_day_cutoff")
    .in("shipper", shippers)
    .eq("day_of_week", dow);
  if (error) {
    // Fail-safe: treat as available if the lookup fails.
    return Object.fromEntries(shippers.map((s) => [s, evaluateScheduleStatus(null, now)]));
  }
  const byShipper = new Map<string, ShipperScheduleRow>();
  for (const r of data ?? []) byShipper.set(r.shipper, r as ShipperScheduleRow);
  return Object.fromEntries(shippers.map((s) => [s, evaluateScheduleStatus(byShipper.get(s) ?? null, now)]));
}
