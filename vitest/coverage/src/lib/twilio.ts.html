 * The fallback path is intentional during the rollout window — better to
 * deliver a slightly buyback-flavoured message than no message. Once Meta
 * approves the dedicated CONVERSION_* templates, populate the SIDs above
 * and this helper transparently switches.
 *
 * Variable shapes for fallback pairs MUST already match between the two
 * templates — the code passes the same `variables` map to whichever SID
 * wins. Compatible pairs as drafted above:
 *   CONVERSION_PICKUP_ARRANGED  ↔ BUYBACK_PICKUP_ARRANGED  ({1: ref, 2: grams, 3: url})
 *   CONVERSION_PICKED_UP        ↔ BUYBACK_PICKED_UP        ({1: grams, 2: ref})
 *   CONVERSION_RECEIVED         ↔ BUYBACK_RECEIVED         ({1: ref, 2: grams})
 *   CONVERSION_REJECTED         ↔ BUYBACK_REJECTED         ({1: ref, 2: reason})
 *
 * The four conversion-only templates (CONFIRMATION, APPROVED, DISPATCHED,
 * DELIVERED) have no compatible buyback fallback. When the conversion SID
 * is empty, callers must pass an explicit fallback (or `null` to skip).
 */
export function templateOrFallback(specific: string, fallback: string): string {
  return specific && specific.trim().length > 0 ? specific : fallback;
}

// ── Environment gating (A-band #6 Phase 3) ────────────────────────────────────
//
// Two related protections for non-prod environments:
//
// 1. `[DEV] ` source-prefix on free-form bodies (`sendWhatsApp`) and on the
//    `source` field of `sendAdminAlert`. Templates can't be modified post-
//    approval, so the source field is where the dev tag surfaces for admins.
//
// 2. `ALLOWED_TEST_PHONES` customer-message gate: in dev, customer-bound
//    recipients are dropped unless explicitly whitelisted. Admin recipients
//    (matching any phone in `COMPLIANCE_ALERT_PHONE`) always pass through —
//    Mukesh still gets dev-mode alerts on his admin number, prefixed `[DEV]`.
//
// `ENV_NAME` resolution: explicit env var wins; otherwise we derive from
// NODE_ENV — `production` → `"prod"`, else `"dev"`. This prevents both failure
// modes: Vercel prod (NODE_ENV=production) defaults to ungated even if
// ENV_NAME isn't set; Codespaces / local dev (NODE_ENV=development) defaults
// to gated, so an accidental `npm run dev` can never blast a real customer.
function envName(): string {
  const explicit = process.env.ENV_NAME?.trim().toLowerCase();
  if (explicit) return explicit;
  return process.env.NODE_ENV === "production" ? "prod" : "dev";
}
 
function isDev(): boolean {
  return envName() !== "prod";
}

/**
 * In dev, drop customer-bound recipients that aren't on the test allow-list.
 * Admin recipients (matching COMPLIANCE_ALERT_PHONE) always pass through.
 * In prod, all recipients pass through unchanged.
 *
 * Logs blocked recipients via console.warn so the gate's effect is visible
 * during local development without spamming structured logs.
 */
function gateRecipientsForEnv(recipients: string[]): string[] {
  if (!isDev()) return recipients;
  const admins  = adminPhoneSet();
  const testers = allowedTestPhoneSet();
  const allowed: string[] = [];
  const blocked: string[] = [];
  for (const r of recipients) {
    const bare = bareE164(r);
    if (admins.has(bare) || testers.has(bare)) {
      allowed.push(r);
    } else {
      blocked.push(r);
    }
  }
  if (blocked.length > 0) {
    console.warn(
      `[twilio] ENV_NAME=${envName()} — blocked ${blocked.length} customer recipient(s) ` +
      `not in ALLOWED_TEST_PHONES: ${blocked.join(", ")}`
    );
  }
  return allowed;
}

/**

 * Accepts a single E.164 phone OR a comma-separated list — sends to each.
 * @param to    E.164 phone (e.g. "+6591234567") or "+1...,+65..." for fan-out
 * @param body  Message text (WhatsApp markdown supported)
 */
export async function sendWhatsApp(to: string, body: string): Promise<void> {
  const from = cleanPhone(process.env.TWILIO_WHATSAPP_FROM, "TWILIO_WHATSAPP_FROM");
  const recipients = gateRecipientsForEnv(parseRecipientList(to, "sendWhatsApp(to)"));
  if (recipients.length === 0) return;
  const taggedBody = devTag(body);
  for (const r of recipients) {
    const normalised = r.startsWith("whatsapp:") ? r : `whatsapp:${r}`;
    await getClient().messages.create({ from, to: normalised, body: taggedBody });
  }
}
 
/**

 * Accepts a single E.164 phone OR a comma-separated list — sends to each.
 * Returns the LAST recipient's Twilio message SID for delivery-status polling
 * (e.g. the OTP flow, which always passes a single phone).
 *
 * @param to          E.164 phone or comma-separated list
 * @param contentSid  Template SID (use TEMPLATE_* constants above)
 * @param variables   Template variable map e.g. { "1": "John", "2": "ABC123" }
 */
export async function sendWhatsAppTemplate(
  to: string,
  contentSid: string,
  variables: Record<string, string>,
  mediaUrl?: string
): Promise<{ sid: string } | null> {
  const from = cleanPhone(process.env.TWILIO_WHATSAPP_FROM, "TWILIO_WHATSAPP_FROM");
  const recipients = gateRecipientsForEnv(parseRecipientList(to, "sendWhatsAppTemplate(to)"));
  if (recipients.length === 0) return null;
  let lastResult: { sid: string } | null = null;
  for (const r of recipients) {
    const normalised = r.startsWith("whatsapp:") ? r : `whatsapp:${r}`;
    const msg = await getClient().messages.create({
      from,
      to:               normalised,
      contentSid,
      contentVariables: JSON.stringify(variables),
      ...(mediaUrl ? { mediaUrl: [mediaUrl] } : {}),
    });
    lastResult = { sid: msg.sid };
  }
  return lastResult;
}

export async function sendAdminAlert(

  source:   string,
  detail:   string,

  const ts = new Date().toLocaleString("en-SG", {
