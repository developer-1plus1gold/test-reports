"use client";
 
import { create } from "zustand";
import { persist } from "zustand/middleware";
import { CURRENCIES, type CurrencyConfig } from "@/lib/currencies";
 
// Re-export so existing client callers (`import { CURRENCIES } from "@/store/currencyStore"`)
// keep working without churn. New code — especially server-side modules
// like generateNewsletterPdf — should import directly from
// "@/lib/currencies" to avoid pulling in the "use client" boundary.
export { CURRENCIES };
export type { CurrencyConfig };
 
// No hardcoded fallback rates table. The persist middleware preserves
// the last successfully-fetched rates across sessions, which is the
// only "fallback" we keep — fresh data on first ever visit, recent
// real data on subsequent visits even if a fetch fails momentarily.
//
// On a brand-new browser with no persisted rates AND a failing initial
// fetch, `rates` stays empty {} — display layers (useFmt, useRate)
// detect this and surface "Loading…" instead of fabricating numbers
// from a stale static table. Wrong numbers are worse than honest
// "we don't know yet".
 
interface CurrencyState {
  currency: string;
  rates: Record<string, number>;
  /** True after rates have been fetched at least once this tab session.
   *  Used to gate "are these live rates or stale fallbacks?" — surfaces
   *  inflict (e.g. AI cart cards) trigger a fetch when this is false. */
  ratesFresh: boolean;
  setCurrency: (c: string) => void;
  setRates: (r: Record<string, number>) => void;
  /** Fetch live FX rates from /api/fx-rates. Idempotent within a tab
   *  session — second call is a no-op while the first is in flight or
   *  has succeeded. Surfaces that depend on accurate cross-currency
   *  display (AI cart cards, calculator option cards) call this on
   *  mount; without it, the persisted store may render fallback rates
   *  that diverge from server-computed values, producing visible
   *  number mismatches between cards and AI narration. */
  fetchRates: () => Promise<void>;
}
 
let ratesInflight: Promise<void> | null = null;
 
export const useCurrencyStore = create<CurrencyState>()(
  persist(
    (set, get) => ({
      currency: "SGD",
      // Empty by default. Persist middleware re-hydrates last-known
      // rates from localStorage on subsequent loads. SGD always resolves
      // (the rate is 1 by definition) — non-SGD displays render
      // "Loading…" until either persisted or live rates land.
      rates: {},
      ratesFresh: false,
      setCurrency: (currency) => set({ currency }),
      setRates: (rates) => set({ rates, ratesFresh: true }),
      fetchRates: async () => {
        if (get().ratesFresh) return;
        if (ratesInflight) return ratesInflight;
        ratesInflight = (async () => {
          try {
            const res  = await fetch("/api/fx-rates", { cache: "no-store" });
            if (!res.ok) return;
            const data = await res.json() as { rates?: Record<string, number> };
            if (data.rates) set({ rates: data.rates, ratesFresh: true });
          } catch { /* keep fallback */ }
          finally  { ratesInflight = null; }
        })();
        return ratesInflight;
      },
    }),
    {
      name: "1plus1gold-currency",
      // Persist selection + last-known rates so switching currency is instant
      partialize: (state) => ({ currency: state.currency, rates: state.rates }),
    }
  )
);
 
/**

 * Components using this will re-render automatically when currency or
 * rates change. When the active non-SGD rate is unavailable (no persisted
 * rate, no fresh fetch yet) the formatter returns "Loading…" rather than
 * fabricating a number — better to surface honest unavailability than to
 * silently emit SGD digits with an INR symbol.
 */
export function useFmt() {
  const currency = useCurrencyStore((s) => s.currency);
  const rates    = useCurrencyStore((s) => s.rates);
  const cfgEntry = CURRENCIES[currency];
  const cfg      = cfgEntry ?? CURRENCIES.SGD;
  // Three resolution branches:
  //   • SGD or an UNKNOWN currency code → rate=1 (defensive fallback,
  //     keeps a misconfigured currency from breaking display)
  //   • a known non-SGD currency with a stored rate → use that rate
  //   • a known non-SGD currency without a stored rate → "Loading…"
  //     (rates haven't loaded yet — better than showing SGD digits
  //     mislabelled with the wrong currency symbol)
  const rate = (currency === "SGD" || !cfgEntry)
    ? 1
    : rates[currency];

/** Returns the BCP 47 locale for a currency code (falls back to "en-US"). */
export function getCurrencyLocale(code: string): string {
  return CURRENCIES[code]?.locale ?? "en-US";
}
 
/** Returns the ISO flag code for the currently selected currency. */
export function useFlag() {
  const currency = useCurrencyStore((s) => s.currency);
  return CURRENCIES[currency]?.flag ?? "sg";
}
 
/**
 * useRate — returns the current SGD → selected currency multiplier.
 * Re-renders when currency or rates change. Falls back to 1 if the
 * rate isn't loaded yet — callers doing math should be paired with
 * useRatesAvailable() if they need to gate UI on rate readiness;
 * useFmt() already surfaces "Loading…" in that case for display. */
export function useRate(): number {
  const currency = useCurrencyStore((s) => s.currency);
  const rates    = useCurrencyStore((s) => s.rates);
  if (currency === "SGD") return 1;
  return rates[currency] ?? 1;
}
 
/**
 * useRatesAvailable — true when the active currency has a usable rate.
 * Components that want to gate entire panels (e.g. "wait for rates
 * before showing prices at all") use this; the alternative is to
 * just call useFmt() and let "Loading…" appear inline per figure. */
export function useRatesAvailable(): boolean {
  const currency = useCurrencyStore((s) => s.currency);
  const rates    = useCurrencyStore((s) => s.rates);
  return currency === "SGD" || rates[currency] !== undefined;
}
