/**
 * MarketingPlanner — coordinator agent.
 *
 * Reads the rolling 4-week window of marketing_content_pieces + the
 * latest Listener performance signals. Produces a weekly plan: which
 * pieces need brief generation this week, which are stuck, what
 * Listener signals should bias next week's content.
 *
 * Fires:
 *   - Cron: every Sunday 20:00 SGT — `pollAndDispatch()` is called by
 *     /api/cron/marketing-weekly-cycle which we wire in vercel.json.
 *   - Event-driven: after any downstream agent completes, the agent
 *     route invokes `reevaluate()` to surface stuck pieces.
 *   - Manual: Control Tower "Run Planner" button.
 *
 * Tier: fast. Planning is pattern-matching over a known-shape state
 * (the calendar), not generative reasoning over open-ended inputs.
 */
 
import { BaseAgent } from "../../base-agent";
import type { ModelTier } from "../../registry";
import { marketingAppFacts, APP_FACTS_BASE, type ContentPieceRow } from "./shared";
 
export interface PlannerInput {
  /** Pieces from the past 14 days — for arc-awareness, NOT action. */
  recentPieces:          ContentPieceRow[];
  /** Pieces in the next 14 days — the actionable focus window. */
  upcomingPieces:        ContentPieceRow[];
  /** Pieces in weeks 3–6 out — visibility only, prevents same-shot collisions. */
  horizonPieces:         ContentPieceRow[];
  /** Listener's latest weekly performance summary (signals to bias next week). */
  recentSignals?:        Array<{ signal: string; weight: number; pieceId: string | null }>;
  /** "what" question — drives the plan output shape. */
  mode:                  "weekly_cycle" | "stuck_check";
}
 
export interface PlannerOutput {
  /** Pieces flagged as ready for brief generation in the upcoming window. */
  briefRequests:         Array<{ contentPieceId: string; rationale: string }>;
  /** Pieces stuck >48h with no status advance — escalations. */
  stuckPieces:           Array<{ contentPieceId: string; stuckSince: string; reason: string }>;
  /** Signals for next week's planning cycle. */
  trendInsights:         string[];
  /** 1–2 sentence narrative of where in the 14-week arc we are. */
  arcContext:            string;
  /** Headline summary for the WhatsApp admin alert (≤200 chars). */
  summary:               string;
}
 
export class MarketingPlannerAgent extends BaseAgent {
  readonly name = "marketing_planner";
  readonly tier: ModelTier = "fast";
 
  getRole(): string {
    return [
      "I am MarketingPlanner for 1Plus1Gold.",
      "I own the 14-week Drumbeat calendar's operational tempo.",
      "Every Sunday 8 PM SGT I see THREE windows of the arc:",
      "  - RECENT (everything before today): what's already shipped + performance signals — biases next-week tone.",
      "  - UPCOMING (next 14 days): the action focus. I decide which need briefs generated NOW",
      "    and surface any pieces stuck >48h to the operator.",
      "  - HORIZON (everything beyond next 14 days): visibility for full-arc awareness, so I don't",
      "    request the same shot twice and can flag collisions early.",
      "My decisions act only on UPCOMING. RECENT and HORIZON shape how I describe context",
      "  in the arcContext narrative, but I never propose brief requests for them.",
      "I respect the chakra arc — chakra Reels (Root → Crown) cannot be",
      "  reordered. The arc is the brand's emotional spine.",
      "I do NOT write briefs, captions, or approve anything. I orchestrate.",
    ].join("\n");
  }
 
  async plan(input: PlannerInput): Promise<PlannerOutput> {
    const messages = [
      {
        role: "user" as const,
        content: this.composeUserPrompt(input),
      },
    ];
    const response = await this.runInference({
      runtimeContext: `Today: ${new Date().toISOString().split("T")[0]}`,
      messages,
      maxTokens: 2000,
      appFactsOverride: marketingAppFacts(APP_FACTS_BASE),
    });
    const m = response.text.match(/```json\s*([\s\S]*?)\s*```/);
    if (!m) throw new Error("Planner output missing JSON block. Raw: " + response.text.slice(0, 300));
    return JSON.parse(m[1]) as PlannerOutput;
  }
 
  private composeUserPrompt(input: PlannerInput): string {
    const lines: string[] = [];
    lines.push(`MODE: ${input.mode}`);
    lines.push("");
    const renderPiece = (p: ContentPieceRow) => {
      lines.push(`  ${p.id} — W${p.week_number} ${p.post_date} ${p.post_day} — ${p.theme}`);
      lines.push(`    register=${p.register} format=${p.format} chakra=${p.chakra ?? "-"} status=${p.status}`);
    };
 
    lines.push(`RECENT — everything before today (${input.recentPieces.length} pieces, arc-awareness ONLY, do not propose briefs for these):`);
    if (input.recentPieces.length === 0) lines.push("  (none — calendar may not have launched yet)");
    else input.recentPieces.forEach(renderPiece);
 
    lines.push("");
    lines.push(`UPCOMING — next 14 days (${input.upcomingPieces.length} pieces — THIS IS YOUR ACTION SCOPE):`);
    if (input.upcomingPieces.length === 0) lines.push("  (none — flag as stuck/calendar-gap)");
    else input.upcomingPieces.forEach(renderPiece);
 
    lines.push("");
    lines.push(`HORIZON — everything beyond next 14 days (${input.horizonPieces.length} pieces, visibility ONLY for collision-avoidance):`);
    if (input.horizonPieces.length === 0) lines.push("  (none — end of calendar approaching)");
    else input.horizonPieces.forEach(renderPiece);
 
    if (input.recentSignals?.length) {
      lines.push("");
      lines.push(`RECENT LISTENER SIGNALS:`);
      for (const s of input.recentSignals) {
        lines.push(`  [w=${s.weight.toFixed(2)}] ${s.signal}${s.pieceId ? ` (piece ${s.pieceId})` : ""}`);
      }
    }
 
    lines.push("");
    lines.push(`OUTPUT FORMAT — wrap in \`\`\`json … \`\`\`:`);
    lines.push("```json");
    lines.push(`{`);
    lines.push(`  "briefRequests": [ { "contentPieceId": "uuid (must be from UPCOMING)", "rationale": "why this week" } ],`);
    lines.push(`  "stuckPieces":   [ { "contentPieceId": "uuid", "stuckSince": "ISO date", "reason": "..." } ],`);
    lines.push(`  "trendInsights": [ "≤2-sentence insight for next-week bias" ],`);
    lines.push(`  "arcContext":    "1-2 sentence narrative of where in the 14-week arc we are — what just shipped + what's next + any horizon notes",`);
    lines.push(`  "summary":       "≤200 char headline for admin WhatsApp alert — should reference the arc, not just brief counts"`);
    lines.push(`}`);
    lines.push(`\`\`\``);
    return lines.join("\n");
  }
}
