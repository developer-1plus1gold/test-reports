/**
 * Short-lived HMAC tokens for public PDF endpoints.
 *
 * Tokens are generated at send time and embedded as a `?t=` query param in
 * the WhatsApp template variable. The PDF route REQUIRES a valid token —
 * requests without a token are rejected with 401.
 *
 * Token format: base64url(JSON(payload)).hmac_signature
 *   - payload includes: userId, resource, resourceId, issuedAt, expiresAt
 *   - signature: HMAC-SHA256(PDF_TOKEN_SECRET, encodedPayload).hex().slice(0, 32)
 *   - default expiry: 1 hour
 *
 * Uses PDF_TOKEN_SECRET env var (separate from ADMIN_SECRET and INTERNAL_API_KEY).
 */
 
import { createHmac, timingSafeEqual } from "crypto";
 
export interface PdfTokenPayload {
  userId:     string;
  resource:   "receipt" | "history" | "agreement" | "newsletter";
  /** For "newsletter": resourceId = "{customerId}:{year}:{month}" so a
   *  newsletter token is bound to both the customer AND the issue period. */
  resourceId: string;
  issuedAt:   number;   // unix seconds
  expiresAt:  number;   // unix seconds
}
 
function b64url(s: string): string {
  return Buffer.from(s).toString("base64url");
}
 
function fromb64url(s: string): string {
  return Buffer.from(s, "base64url").toString("utf8");
}
 
function signPayload(encodedPayload: string): string {
  const key = process.env.PDF_TOKEN_SECRET?.trim() ?? "";
  if (!key) return "";
  return createHmac("sha256", key).update(encodedPayload).digest("hex").slice(0, 32);
}
 
/**
 * Generates a signed, time-limited PDF access token.
 * Returns empty string if PDF_TOKEN_SECRET is not configured.
 *
 * Default expiry is 15 minutes (MED-008 from security audit 2026-05-03).
 * Templates resolve the URL once at Twilio→Meta send time; Meta caches the
 * media on its CDN. We therefore only need to keep the URL valid long
 * enough to absorb queue delay + retries (~minutes), not the full hour
 * the original design allowed. Tighter window cuts the token-leak blast
 * radius (e.g. URL exposed via dev tools / logs) by ~4×.
 */
export function generatePdfToken(
  payload: Omit<PdfTokenPayload, "issuedAt" | "expiresAt">,
  expiresInSeconds = 900
): string {
  const now = Math.floor(Date.now() / 1000);
  const full: PdfTokenPayload = {
    ...payload,
    issuedAt:  now,
    expiresAt: now + expiresInSeconds,
  };
  const encodedPayload = b64url(JSON.stringify(full));
  const sig = signPayload(encodedPayload);
  if (!sig) return "";
  return `${encodedPayload}.${sig}`;
}
