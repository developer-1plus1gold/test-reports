/**
 * Shared helpers for WebAuthn / Passkey routes.
 *
 * Resolves the RP configuration from the request itself so we don't have
 * to maintain three different env vars per environment (codespace dev,
 * Vercel previews, prod). What we need:
 *
 *   rpID    — the registrable origin host. e.g. "1plus1gold.com" or
 *             "glowing-robot-…app.github.dev". NO scheme, NO port.
 *   origin  — full URL prefix as the browser sees it. e.g.
 *             "https://1plus1gold.com" or "http://localhost:3000".
 *   rpName  — human-readable name shown in OS/browser passkey prompts.
 *
 * Both rpID and origin must match what the browser computed at register
 * time, exactly. Mismatch → @simplewebauthn/server rejects with "RP ID
 * hash does not match" or "Unexpected authentication response origin".
 *
 * We support multiple legitimate origins per deploy (prod main + naked
 * domain redirect, codespace flavor, prebuild preview URLs) by reading
 * the live request and trusting it — the request can ONLY come through
 * Vercel's edge after the host header is canonicalised.
 */
 
import type { NextRequest } from "next/server";
 
export const RP_NAME = "1Plus1Gold";
 
export interface RpConfig {
  rpID:     string;
  origin:   string;
  rpName:   string;
}
 
export function getRpConfig(req: NextRequest): RpConfig {
  // nextUrl.origin already contains the canonical scheme + host + port from
  // the proxy / edge. Use that as the source of truth — never hand-derive
  // from x-forwarded-* headers (spoofable on misconfigured deployments).
  const url    = req.nextUrl;
  const origin = url.origin;
  const rpID   = url.hostname;
 
  return { rpID, origin, rpName: RP_NAME };
}
 
/**
 * Cookie name we use to ferry the per-flow challenge between the begin
 * and complete API calls. The challenge is a one-shot value the
 * browser signs over; the server must compare what it issued against
 * what comes back. httpOnly + signed-by-the-app-secret would be ideal,
 * but the simplest robust pattern is: short-lived (5 min), httpOnly,
 * Secure (when not on http://localhost), Strict same-site.
 */
export const CHALLENGE_COOKIE_REGISTER = "__pkr";
export const CHALLENGE_COOKIE_SIGNIN   = "__pks";
 
/** Five-minute TTL on challenge cookies — the user has to complete the
 *  OS prompt within this window. Plenty of headroom for biometric +
 *  cross-device hybrid flows. */
export const CHALLENGE_TTL_SECONDS = 300;
 
/** Convert a base64url-encoded credential ID (what the browser sends
 *  back) to/from the bytea storage format we use server-side. */
export function base64urlToBuffer(s: string): Buffer {
  // base64url → base64 → Buffer
  const b64 = s.replace(/-/g, "+").replace(/_/g, "/");
  const pad = b64.length % 4 === 0 ? "" : "=".repeat(4 - (b64.length % 4));
  return Buffer.from(b64 + pad, "base64");
}
 
export function bufferToBase64url(b: Buffer | Uint8Array): string {
  return Buffer.from(b)
    .toString("base64")
    .replace(/\+/g, "-")
    .replace(/\//g, "_")
    .replace(/=+$/, "");
}
