/**
 * NFC serial — format: 1G-XXXX-XXXX
 *
 * Encoding: 8 chars from a 30-char alphabet (uppercase A–Z + 2–9, minus the
 * visually ambiguous I, O, L, 1, 0). 30^8 ≈ 6.6 × 10¹¹ combinations — way
 * more than we'll ever issue, low enough to encode in 8 hand-typeable chars.
 *
 * The "1G-" prefix is brand metadata and is fixed; only the trailing 8
 * characters carry entropy. Don't shorten the prefix without coordinating
 * with the regex check constraint in 20260430_nfc_serials.sql.
 */
import { randomBytes } from "node:crypto";
 
const ALPHABET = "ABCDEFGHJKMNPQRSTUVWXYZ23456789"; // length 30
const SERIAL_REGEX = /^1G-[A-HJ-NP-Z2-9]{4}-[A-HJ-NP-Z2-9]{4}$/;
 
/** Generate a single serial. Caller must check uniqueness against the DB. */
export function generateSerial(): string {
  // Use rejection sampling so the alphabet distribution is exactly uniform
  // (modulo bias would skew letters at the start of the alphabet very
  // slightly — invisible at our scale, but unnecessary).
  const out: string[] = [];
  while (out.length < 8) {
    const buf = randomBytes(16);
    for (const byte of buf) {
      if (byte >= ALPHABET.length * Math.floor(256 / ALPHABET.length)) continue;
      out.push(ALPHABET[byte % ALPHABET.length]);
      if (out.length === 8) break;
    }
  }
  return `1G-${out.slice(0, 4).join("")}-${out.slice(4, 8).join("")}`;
}
 
/** Generate a batch — caller still has to insert + handle collision retries. */
export function generateSerialBatch(n: number): string[] {
  const seen = new Set<string>();
  while (seen.size < n) seen.add(generateSerial());
  return [...seen];
}
 
/** Format check — does NOT confirm existence in DB. */
export function isValidSerialFormat(serial: string): boolean {
  return SERIAL_REGEX.test(serial);
}
 
/** Normalise user input — strip whitespace, uppercase, accept missing
 *  dashes (so a customer can type `1GA3F9K2L7` and we still resolve it). */
export function normaliseSerial(input: string): string | null {
  const cleaned = input.replace(/[\s-]/g, "").toUpperCase();
  // Either 10 chars (1G + 8) or already-prefixed without dashes
  if (cleaned.length !== 10 || !cleaned.startsWith("1G")) return null;
  const body = cleaned.slice(2);
  const formatted = `1G-${body.slice(0, 4)}-${body.slice(4, 8)}`;
  return isValidSerialFormat(formatted) ? formatted : null;
}
