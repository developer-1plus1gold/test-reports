/**
 * Channel-router helpers — one per notification kind.
 *
 * Each helper looks up the customer's phone + email and fires whichever
 * channels are populated. Result: phone-OTP customers get WhatsApp (as
 * before), Apple/Google-sign-in customers get email, customers with
 * both get both (redundancy at no extra effort).
 *
 * Use these from any server route that wants to notify a customer.
 * The 43+ historical `sendWhatsAppTemplate` call sites are being
 * migrated to these helpers incrementally — start with the highest-
 * impact ones (OTP, order confirmation, receipt, KYC, delivery), sweep
 * the rest in Day 6.
 *
 * Each call returns { whatsapp, email } sub-results so callers can log
 * partial failures. Soft-fail: failures are logged but never thrown.
 */

// ── Internal: look up channels for a customer ────────────────────────────
type CategoryPrefs = { channels?: string[] };
type Prefs = Record<string, CategoryPrefs>;
type Channels = {
  phone: string | null;
  email: string | null;
  firstName: string | null;
  prefs: Prefs;
};
 
/**
 * Optional notification categories that customers can opt in/out of via
 * /account → Notifications. Mandatory categories (order confirmation,
 * receipt, KYC, security alerts, etc.) are NOT in this set — they always
 * fan out to all available channels regardless of prefs.
 */
const OPTIONAL_CATEGORIES = new Set(["newsletter", "marketing", "gifts"]);
 
function admin() {
  return createClient(process.env.NEXT_PUBLIC_SUPABASE_URL!, process.env.SUPABASE_SERVICE_KEY!, { auth: { persistSession: false } });
}
 
/**
 * Resolve { phone, email, firstName, prefs } for a customer id
 * (= auth.users.id). Phone comes from profiles.phone; email comes from
 * auth.users.email (but synthetic phone-OTP emails like
 * phone_xxx@auth.1plus1gold.com are filtered to null — they're not real
 * inboxes). prefs comes from profiles.notification_prefs (see migration
 * 20260520d for shape + defaults).
 */
export async function getChannels(customerId: string): Promise<Channels> {
  const sb = admin();
  // Run both lookups in parallel.
  const [profileResult, userResult] = await Promise.all([
    sb.from("profiles").select("full_name, phone, notification_prefs").eq("id", customerId).maybeSingle(),
    sb.auth.admin.getUserById(customerId),
  ]);
 
  const phone = profileResult.data?.phone || null;
  const rawEmail = userResult.data?.user?.email ?? null;
  const email = rawEmail && !rawEmail.includes("@auth.1plus1gold.com") ? rawEmail : null;
  const firstName = (profileResult.data?.full_name ?? "").trim().split(/\s+/)[0] || null;
  const prefs = (profileResult.data?.notification_prefs ?? {}) as Prefs;
  return { phone, email, firstName, prefs };
}
 
/**
 * For an OPTIONAL category, return which channels are allowed by the
 * customer's prefs. For MANDATORY categories (anything not in
 * OPTIONAL_CATEGORIES), all channels are allowed.
 */
function allowedChannels(
  channels: Channels,
  category: string | undefined,
): { whatsapp: boolean; email: boolean } {
  if (!category || !OPTIONAL_CATEGORIES.has(category)) {
    // Mandatory — always allow whichever channels the customer has.
    return { whatsapp: true, email: true };
  }
  const list = channels.prefs[category]?.channels ?? [];
  return { whatsapp: list.includes("whatsapp"), email: list.includes("email") };
}
 
// ── Standard result shape ────────────────────────────────────────────────
export type NotifyResult = {
  whatsapp: { ok: boolean; sid?: string | null; reason?: string };
  email:    { ok: boolean; messageId?: string | null; reason?: string };
};
 
function makeResult(): NotifyResult {
  return {
    whatsapp: { ok: false, reason: "no_phone" },
    email:    { ok: false, reason: "no_email" },
  };
}
 
// ── Helper: dispatch a WhatsApp template + an email in parallel ──────────
// `category` (optional): when this notification belongs to an OPTIONAL
// category (newsletter / marketing / gifts), the customer's prefs gate
// which channels actually fire. Mandatory categories pass undefined and
// always fan out to all available channels.
async function dispatch<TVars extends Record<string, string>>(
  channels: Channels,
  wa: { sid: string; vars: TVars } | null,
  em: { subject: string; html: string; text: string; source: string; attachments?: Parameters<typeof sendEmail>[0]["attachments"] } | null,
  category?: string,
): Promise<NotifyResult> {
  const result = makeResult();
  const allowed = allowedChannels(channels, category);
  const tasks: Array<Promise<void>> = [];
 
  if (channels.phone && wa && allowed.whatsapp) {
    tasks.push((async () => {
      try {
        const sent = await sendWhatsAppTemplate(channels.phone!, wa.sid, wa.vars);
        result.whatsapp = { ok: true, sid: sent?.sid ?? null };
      } catch (err) {
        result.whatsapp = { ok: false, reason: err instanceof Error ? err.message : "send_failed" };
      }
    })());
  } else if (channels.phone && wa && !allowed.whatsapp) {
    result.whatsapp = { ok: false, reason: "pref_opted_out" };
  }
 
  if (channels.email && em && allowed.email) {
    tasks.push((async () => {
      const r = await sendEmail({
        to: channels.email!,
        subject: em.subject,
        html: em.html,
        text: em.text,
        source: em.source,
        attachments: em.attachments,
      });
      result.email = r.ok
        ? { ok: true, messageId: r.messageId }
        : { ok: false, reason: r.reason };
    })());
  } else if (channels.email && em && !allowed.email) {
    result.email = { ok: false, reason: "pref_opted_out" };
  }
 
  await Promise.allSettled(tasks);
  return result;
}

/** Receipt PDF — sends the PDF as both WhatsApp media + email attachment.
 *
 *  WhatsApp: the Twilio media template's URL is `/api/orders/receipt-pdf/{{2}}`,
 *  so `{{2}}` literally carries `${orderId}?t=${pdfToken}` — the `?` becomes
 *  the URL's query separator and the route reads `t` from searchParams.
 *
 *  Email: a separate `pdfUrl` is built by receiptEmail() with orderId and
 *  pdfToken as distinct args (token in the query string, not stuffed into
 *  the path). The PDF is also attached when pdfBase64 is provided so the
 *  customer doesn't need to click out — same parity as the WhatsApp media. */
export async function notifyReceipt(
  customerId: string,
  orderRef: string,
  orderId: string,                  // clean UUID
  pdfToken: string | null,          // HMAC token; null = no token gate
  pdfBase64: string | null,         // base64 PDF bytes for email attachment
): Promise<NotifyResult> {
  const ch = await getChannels(customerId);
  const attachments = pdfBase64
    ? [{ filename: `receipt-${orderRef}.pdf`, content: pdfBase64, type: "application/pdf" }]
    : undefined;
  const waOrderIdParam = pdfToken ? `${orderId}?t=${pdfToken}` : orderId;
  return dispatch(
    ch,
    { sid: TEMPLATE_RECEIPT_PDF, vars: { "1": orderRef, "2": waOrderIdParam } },
    {
      ...receiptEmail({ firstName: ch.firstName ?? undefined, orderRef, orderId, pdfToken }),
      source: "receipt_pdf",
      attachments,
    },
  );
}
