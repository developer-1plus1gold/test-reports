/**
 * Tiny request-body validation helper around zod.
 *
 * MED batch 3 (security audit 2026-05-03) — adopt zod on high-touch
 * routes (auth, webhooks, orders, payments) so request bodies have an
 * explicit schema instead of ad-hoc field-by-field nullish checks.
 *
 * Returns a discriminated union so callers can short-circuit with the
 * pre-built NextResponse instead of writing the same try/catch around
 * `await req.json()` everywhere.
 *
 * Default behaviour is `.strip()` (zod default — unknown keys silently
 * dropped). Routes that need strict reject-on-extra-keys behaviour
 * should call `schema.strict()` at definition time. We don't force
 * strict here because some legacy frontends send client-side fields
 * that the server doesn't read; strict on them would be a regression.
 */
 
import { NextResponse } from "next/server";
import { z, type ZodType } from "zod";
 
export type ParseResult<T> =
  | { ok: true;  data: T }
  | { ok: false; res: NextResponse };
 
export async function parseJsonBody<T>(
  req: Request,
  schema: ZodType<T>,
): Promise<ParseResult<T>> {
  let raw: unknown;
  try {
    raw = await req.json();
  } catch {
    return { ok: false, res: NextResponse.json({ error: "Invalid JSON" }, { status: 400 }) };
  }
 
  const result = schema.safeParse(raw);
  if (!result.success) {
    // Surface the first zod issue's message — short, actionable, safe for
    // public consumption (zod messages are user-friendly by default and
    // don't leak server internals).
    const first = result.error.issues[0];
    const message = first?.message ?? "Invalid request body";
    return { ok: false, res: NextResponse.json({ error: message }, { status: 400 }) };
  }
  return { ok: true, data: result.data };
}
 
// ─────────────────────────────────────────────────────────────────────────────
// Shared cart body schema
//
// Three routes (orders/create, payments/hitpay/create, payments/stripe/
// create-intent) accept the same checkout body shape. Defining it once
// avoids drift when fields are added or constraints change. Each route
// extends with its own payment-method narrowing and may add route-
// specific fields via `.merge()`.
//
// `.passthrough()` is intentional — the cart frontend sometimes ships
// transitional UI-only fields ahead of the server consuming them; we
// don't want a typo in *new* checkout-step data to break *existing*
// checkouts. Numeric guards still hold on the fields we DO read.
// ─────────────────────────────────────────────────────────────────────────────
 
export const cartItemSchema = z.object({
  productId: z.string().nullish(),
  grams:     z.number().positive(),
  quantity:  z.number().int().positive().max(500),
}).passthrough();
 
export const cartAccessorySchema = z.object({
  productId: z.string(),
  quantity:  z.number().int().positive(),
}).passthrough();
 
export const cartBodySchema = z.object({
  // Messages chosen to match the legacy error strings the cart frontend +
  // existing test suite already assert against (see b2-payment-routes,
  // c9-payment-routes-early). Don't change without updating those tests.
  items:                z.array(cartItemSchema).min(1, "Cart is empty"),
  accessories:          z.array(cartAccessorySchema).optional(),
  lockedSpotSgd:        z.number({ error: "Invalid locked prices" }).positive("Invalid locked prices"),
  lockedSubtotalSgd:    z.number({ error: "Invalid locked prices" }).positive("Invalid locked prices"),
  shipperType:          z.string().optional(),
  shippingFee:          z.number().nonnegative().optional(),
  deliveryAddress:      z.record(z.string(), z.unknown()).optional(),
  quotationId:          z.string().optional(),
  quotationExpiresAt:   z.string().optional(),
  senderStopId:         z.string().optional(),
  stopId:               z.string().optional(),
  serviceType:          z.string().optional(),
  recipientName:        z.string().optional(),
  recipientPhone:       z.string().optional(),
  paymentMethod:        z.string().optional(),
  promoCode:            z.string().optional(),
  referralCode:         z.string().optional(),
  refundPolicyAccepted: z.boolean().optional(),
}).passthrough();
 
export type CartBody = z.infer<typeof cartBodySchema>;
