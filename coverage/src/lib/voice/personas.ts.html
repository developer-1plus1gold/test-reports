/**
 * Voice persona registry — the "Six Guides".
 *
 * Each guide is named after the word for "gold" in a language our
 * customers speak. The picker shows the name + meaning + capability
 * line so first contact teaches something about gold while building
 * familiarity with the guide.
 *
 * Why "guide" not "advisor": "advisor" is regulated language under
 * MAS (financial advisor / investment advisor). 1Plus1Gold is a
 * precious-metal dealer, not licensed for advice. Guides point at
 * information and facilitate purchases — never recommend financial
 * decisions.
 *
 * Persona keys (voice_onyx etc.) keep the original tts-1-flavoured
 * names for stability across stored customer selections — only the
 * customer-facing display name + metadata changes.
 *
 * Voice catalogue note: the OpenAI Realtime + tts-1 catalogues differ.
 * openaiVoice values are pinned to the intersection (alloy, ash,
 * coral, echo, sage, shimmer) so a single field works in both modes.
 */
 
import type { UserContext } from "@/lib/ai-calculator";
 
export type PersonaKey =
  | "voice_alloy"
  | "voice_echo"
  | "voice_fable"
  | "voice_nova"
  | "voice_onyx"
  | "voice_shimmer";
 
export interface Persona {
  /** Stable storage key — NEVER change without a migration. */
  key:          PersonaKey;
  /** Customer-facing guide name. */
  name:         string;
  /** Language family the name comes from (Latin / Arabic / etc.). */
  language:     string;
  /** One-line meaning shown in the picker, narrated on first-meet. */
  meaning:      string;
  /** What this guide can help with — used in picker + first-meet intro. */
  helpScope:    string;
  /** Voice gender — injected into the prompt so the model self-identifies
   *  correctly if asked (avoids the gpt-realtime-2 quirk of mis-gendering). */
  gender:       "male" | "female";
  /** One-line tone cue injected into the system prompt so the model
   *  shapes its register to match the guide's character. */
  toneCue:      string;
  /** OpenAI voice ID — must be in tts-1 ∩ Realtime intersection. */
  openaiVoice:  string;
  /** Sample sentence played when the customer taps ▶ in the picker.
   *  Same shape the model uses for the first-meet greeting so the
   *  audition IS the brand moment, not a placeholder. */
  auditionScript: string;
}

/** Iteration order for the picker. Aurel anchors the lineup as the
 *  default; the rest follow in a culturally-balanced sequence. */
export const PERSONA_ORDER: PersonaKey[] = [

export const DEFAULT_PERSONA: PersonaKey = "voice_onyx";
 
/** Pre-Phase-2.5 keys → new equivalents. resolvePersona migrates these
 *  silently so customers keep their selection across the upgrade. */
const LEGACY_KEY_MAP: Record<string, PersonaKey> = {
  voice_1: "voice_onyx",
  voice_2: "voice_nova",
};
 
/** Resolve a stored value to a valid persona, migrating legacy keys
 *  and falling back if unknown. */
export function resolvePersona(value: string | null | undefined): Persona {
  if (!value) return PERSONAS[DEFAULT_PERSONA];
  const migrated = (LEGACY_KEY_MAP[value] ?? value) as PersonaKey;
  return PERSONAS[migrated] ?? PERSONAS[DEFAULT_PERSONA];
}
 
/** Per-guide block of the system prompt — tone + intro line.
 *
 *  Per the production-voice playbook (May 2026): "tiny behavioural
 *  constraints, no backstories." We give the model ONE tone-cue line
 *  and ONE intro instruction, no 5-step roleplay structure. The model
 *  delivers the rest from the natural conversational register the
 *  voice itself supplies. */
export function buildGuideBlock(
  persona:  Persona,
  firstMet: boolean,
  userCtx?: UserContext,
): string {
  const pronoun = persona.gender === "male" ? "him / he / his" : "her / she / hers";
 
  const intro = firstMet
    ? `[FIRST UTTERANCE — first meeting, MUST follow this order]
1. Open by saying your name verbatim: "Hello, I'm ${persona.name}." (or a register-appropriate greeting + your name — but your name MUST appear in the first sentence).
2. ONE short line on what your name means: ${persona.meaning}
3. ONE open question inviting them in.
Three short sentences total. Never skip the name. After this opener, return to one-sentence default.`
    : `[FIRST UTTERANCE — returning customer]
A short warm greeting. Mention you're ${persona.name} in the opener. One or two sentences. Reference spot or their accumulation only if it adds something.`;
 
  const userLine = (userCtx?.ownedGrams ?? 0) > 0
    ? `Customer: ${userCtx!.ownedGrams!.toFixed(2)}g accumulated, working toward 50g.`
    : `Customer: anonymous session, no purchase history.`;
 
  return `[YOU ARE ${persona.name.toUpperCase()} — ${persona.gender} voice; refer to yourself with ${pronoun}]
${persona.toneCue}
 
${intro}
 
${userLine}`;
}
