/**
 * Unified push fan-out helper.
 *
 * Given a user_id, sends a notification to every active push subscription
 * (iOS APNs + web/TWA Web Push) registered against that user. Failures on
 * one device don't block the others — each transport is fire-and-soft-fail.
 *
 * APNs setup (.p8 key auth flow):
 *   - APNS_KEY_P8     — contents of the .p8 file (full PEM, no surrounding quotes)
 *   - APNS_KEY_ID     — 10-char identifier from Apple Dev portal (e.g. F5P8L9DV3A)
 *   - APPLE_TEAM_ID   — 10-char team ID (8XG3SUZ36J)
 *   - APNS_BUNDLE_ID  — com.oneplusonegold.ios
 *   - APNS_PRODUCTION — "true" hits api.push.apple.com (prod), "false" hits sandbox.
 *                       Defaults to true since TestFlight + App Store builds use prod APNs.
 *
 * Web Push (VAPID):
 *   - VAPID_PUBLIC_KEY  — base64url-encoded P-256 public key (also shipped to clients)
 *   - VAPID_PRIVATE_KEY — base64url-encoded P-256 private key (server-only)
 *   - VAPID_SUBJECT     — mailto: or https: URL identifying us to push servers
 *                         (defaults to mailto:support@1plus1gold.com)
 *
 * Both transports degrade gracefully: if env is missing, sending is a no-op
 * and the caller logs at debug. This lets us ship the table + routes before
 * Mukesh adds the secrets to Vercel, without breaking prod.
 */

export interface PushPayload {
  title:  string;
  body:   string;
  /** Optional deep-link path. iOS deep-link wiring is a v1.1 task on the
   *  native side; web/TWA can pass-through to the SW which navigates. */
  route?: string;
  /** Optional badge count for iOS (1 = unread). */
  badge?: number;
  /** Optional override; defaults to "default" (system tone). */

}
 
interface PushRow {
  id:       string;
  platform: "ios" | "android" | "web";
  token:    string | null;
  endpoint: string | null;
  p256dh:   string | null;
  auth:     string | null;
}
 
/**
 * Send a push to every active subscription for the given user. Returns the
 * counts so callers can log meaningful telemetry.
 */
export async function sendPushToUser(
  userId: string,
  payload: PushPayload,
): Promise<{ attempted: number; succeeded: number; revoked: number }> {
  const { data: rows, error } = await sb()
    .from("push_subscriptions")
    .select("id, platform, token, endpoint, p256dh, auth")
    .eq("user_id", userId)
    .is("revoked_at", null);
 
  if (error || !rows) {
    console.warn("[push/send] fetch subscriptions failed:", error?.message);
    return { attempted: 0, succeeded: 0, revoked: 0 };
  }
 
  let succeeded = 0;
  let revoked = 0;
  const subs = rows as unknown as PushRow[];
 
  await Promise.all(
    subs.map(async (row) => {
      try {
        if (row.platform === "web") {
          await sendWebPush(row, payload);
          succeeded++;
        } else if (row.platform === "ios") {
          await sendApns(row, payload);
          succeeded++;
        }
        // android intentionally TODO — no native Android shell shipped yet.
      } catch (err) {
        const e = err as { statusCode?: number; status?: number; message?: string };
        const code = e?.statusCode ?? e?.status ?? 0;
        // 410 Gone (web push) / 410 BadDeviceToken (APNs HTTP/2) means the
        // subscription is permanently invalid. Soft-delete so we stop hitting it.
        if (code === 410 || /BadDeviceToken|Unregistered/i.test(e?.message || "")) {
          // eslint-disable-next-line @typescript-eslint/no-explicit-any
          await (sb().from("push_subscriptions") as any)
            .update({ revoked_at: new Date().toISOString() })
            .eq("id", row.id);
          revoked++;
        } else {
          console.warn(`[push/send] ${row.platform} failed for ${row.id}:`, e?.message);
        }
      }
    }),
  );
 
  return { attempted: subs.length, succeeded, revoked };
}
 
async function sendWebPush(row: PushRow, payload: PushPayload): Promise<void> {
  if (!webPushReady) throw new Error("VAPID not configured");
  if (!row.endpoint || !row.p256dh || !row.auth) throw new Error("invalid web sub");
  await webpush.sendNotification(
    {
      endpoint: row.endpoint,
      keys: { p256dh: row.p256dh, auth: row.auth },
    },
    JSON.stringify({
      title: payload.title,
      body:  payload.body,
      route: payload.route,
    }),
    { TTL: 3600 },
  );
}
 
// APNs send: builds a JWT (ES256 over .p8) and POSTs to api.push.apple.com.
// We use HTTP/2 directly via node:http2 instead of pulling in `apn` to keep
// the dep footprint minimal — APNs HTTP/2 is straightforward and the JWT
// can be cached for 50 minutes (Apple rejects re-issuance < 20 min apart).
import http2 from "node:http2";
import { createSign } from "node:crypto";
 
let _apnsJwt: { token: string; issuedAt: number } | null = null;
 
function getApnsJwt(): string {
  const keyP8  = process.env.APNS_KEY_P8;
  const keyId  = process.env.APNS_KEY_ID;
  const teamId = process.env.APPLE_TEAM_ID;
  if (!keyP8 || !keyId || !teamId) throw new Error("APNs not configured");
 
  const now = Math.floor(Date.now() / 1000);
  // Re-use JWT for 50 minutes (Apple TTL is 1h; rotate before edge).
  if (_apnsJwt && now - _apnsJwt.issuedAt < 50 * 60) return _apnsJwt.token;
 
  const header = Buffer.from(JSON.stringify({ alg: "ES256", kid: keyId })).toString("base64url");
  const claims = Buffer.from(JSON.stringify({ iss: teamId, iat: now })).toString("base64url");
  const unsigned = `${header}.${claims}`;
 
  // .p8 file is a PKCS#8 PEM. Node's createSign handles it natively.
  const signer = createSign("SHA256");
  signer.update(unsigned);
  signer.end();
  // Apple wants the signature as P1363 (r||s, 64 bytes), not DER.
  const der = signer.sign({ key: keyP8, dsaEncoding: "ieee-p1363" });
  const token = `${unsigned}.${der.toString("base64url")}`;
  _apnsJwt = { token, issuedAt: now };
  return token;
}
 
async function sendApns(row: PushRow, payload: PushPayload): Promise<void> {
  if (!row.token) throw new Error("no APNs token");
  const bundleId = process.env.APNS_BUNDLE_ID || "com.oneplusonegold.ios";
  const prod     = (process.env.APNS_PRODUCTION || "true") !== "false";
  const host     = prod ? "api.push.apple.com" : "api.sandbox.push.apple.com";
 
  const jwt = getApnsJwt();
 
  const body = JSON.stringify({
    aps: {
      alert: { title: payload.title, body: payload.body },
      badge: payload.badge,
      sound: payload.sound || "default",
    },
    ...(payload.route ? { route: payload.route } : {}),
  });
 
  await new Promise<void>((resolve, reject) => {
    const client = http2.connect(`https://${host}`);
    const req = client.request({
      ":method":         "POST",
      ":path":           `/3/device/${row.token}`,
      authorization:     `bearer ${jwt}`,
      "apns-topic":      bundleId,
      "apns-push-type":  "alert",
      "apns-priority":   "10",
      "content-type":    "application/json",
      "content-length":  Buffer.byteLength(body),
    });
 
    let status = 0;
    let respBody = "";
    req.on("response", (h) => { status = Number(h[":status"]) || 0; });
    req.on("data",     (chunk) => { respBody += chunk; });
    req.on("end", () => {
      client.close();
      if (status >= 200 && status < 300) resolve();
      else {
        const err = new Error(`APNs ${status} ${respBody}`) as Error & { statusCode: number };
        err.statusCode = status;
        reject(err);
      }
    });
    req.on("error", (e) => { client.close(); reject(e); });
    req.write(body);
    req.end();
  });
}
