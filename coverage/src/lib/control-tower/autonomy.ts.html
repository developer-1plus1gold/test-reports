/**
 * Autonomy ladder helpers — used by the graduation cron and by the
 * decision-PATCH route that applies a promotion when admin approves one.
 *
 * Ladder (lowest → highest trust):
 *   approve  — every decision needs admin tap
 *   notify   — agent acts; admin sees but doesn't tap
 *   auto     — agent acts; logged only
 *   trusted  — same as auto, no further graduation possible (terminal)
 *
 * A graduation proposal is itself an `agent_decisions` row with
 * `action_payload.type = 'autonomy_promotion'` at severity='approve' —
 * admin treats it like any other queue item, taps approve, and the PATCH
 * route's side-effect calls `applyAutonomyPromotion()` to bump the agent.
 */
 
import type { SupabaseClient } from "@supabase/supabase-js";
 
export type AutonomyLevel = "approve" | "notify" | "auto" | "trusted";
 
export const AUTONOMY_LADDER: AutonomyLevel[] = ["approve", "notify", "auto", "trusted"];
 
export function nextAutonomyLevel(current: AutonomyLevel): AutonomyLevel | null {
  const idx = AUTONOMY_LADDER.indexOf(current);
  if (idx < 0 || idx >= AUTONOMY_LADDER.length - 1) return null;
  return AUTONOMY_LADDER[idx + 1];
}
 
export interface GraduationCriteria {
  /** Minimum clean decisions at the current level before graduation. */
  min_decisions?: number;
  /** Minimum days at the current level before graduation. */
  min_days?:      number;
}
 
export interface PromotionPayload {
  type:             "autonomy_promotion";
  agent_name:       string;
  current_level:    AutonomyLevel;
  proposed_level:   AutonomyLevel;
  decisions_count:  number;
  days_at_level:    number;
}
 
/**
 * Apply an approved autonomy_promotion proposal:
 *   • UPDATE ai_agents SET autonomy_level=proposed, autonomy_level_set_at=NOW(), decisions_at_current_level=0
 *   • Insert agent_events row of type 'autonomy_changed'
 *
 * Idempotent — if the agent is already at proposed_level, this is a no-op
 * but still writes the audit event so the trail is honest.
 */
export async function applyAutonomyPromotion(
  sb:        SupabaseClient,
  payload:   PromotionPayload,
  decisionId: string,
): Promise<void> {
  const { error: updErr } = await sb
    .from("ai_agents")
    .update({
      autonomy_level:             payload.proposed_level,
      autonomy_level_set_at:      new Date().toISOString(),
      decisions_at_current_level: 0,
    })
    .eq("name", payload.agent_name);
 
  if (updErr) {
    console.warn(`[autonomy] promotion update failed for ${payload.agent_name}: ${updErr.message}`);
    return;
  }
 
  await sb.from("agent_events").insert({
    agent_name:          payload.agent_name,
    event_type:          "autonomy_changed",
    related_decision_id: decisionId,
    actor:               "human:admin",
    payload: {
      from: payload.current_level,
      to:   payload.proposed_level,
      decisions_count: payload.decisions_count,
      days_at_level:   payload.days_at_level,
    },
  });
}
