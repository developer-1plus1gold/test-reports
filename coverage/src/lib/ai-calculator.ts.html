/** A line item in the simulated cart — same three types the real cart
 *  supports, so a customer asking the AI to price "5g + a 10g bar +
 *  a velvet pouch" gets the same total they'll see at checkout. */

}

// ── Standard-option picker (shared by calculator + AI) ──────────────────────
 
/** Pick the best standard-bar option for a given budget or grams target.
 *  Used by the calculator UI directly, and by the simulate_cart server
 *  routes to auto-pair a standard alternative alongside any curve fill —
 *  so calculator + AI surface the same comparison without duplicating
 *  the matching logic.
 *
 *  Behaviour:
 *    grams mode:
 *      - Exact match (qty 1) preferred
 *      - Otherwise smallest bar ≥ target (qty 1) with a "+Xg more than
 *        asked" note
 *      - If target exceeds every bar, ceil-multiply the largest
 *    budget mode:
 *      - Pick the LARGEST per-unit bar that fits at qty ≥ 1
 *      - Take qty = floor(budget / priceEach)
 *      - Single clean alternative; no mixing sizes
 *
 *  Returns null when no standard bar can be priced for the input
 *  (e.g., budget below the smallest bar's price). */
export function pickStandardOption(
  mode:             "budget" | "grams",
  inputValue:       number,    // grams for grams mode, SGD for budget mode
  spotSgd:          number,
  standardProducts: StandardProductForAI[],
): { items: CartItemInput[]; note: string | null } | null {
  if (standardProducts.length === 0) return null;
 
  if (mode === "grams") {
    const sorted = [...standardProducts].sort((a, b) => a.grams - b.grams);
    const exact  = sorted.find((p) => p.grams === inputValue);
    if (exact) {
      return { items: [{ type: "standard_bar", grams: exact.grams, label: exact.label, quantity: 1 }], note: null };
    }
    const overshoot = sorted.find((p) => p.grams >= inputValue);
    if (overshoot) {
      const delta = overshoot.grams - inputValue;
      return {
        items: [{ type: "standard_bar", grams: overshoot.grams, label: overshoot.label, quantity: 1 }],
        note:  `${delta.toFixed(2)}g more than asked`,
      };
    }
    const largest = sorted[sorted.length - 1];
    const qty     = Math.ceil(inputValue / largest.grams);
    const totalG  = qty * largest.grams;
    const delta   = totalG - inputValue;
    return {
      items: [{ type: "standard_bar", grams: largest.grams, label: largest.label, quantity: qty }],
      note:  delta > 0 ? `${delta.toFixed(2)}g more than asked` : null,
    };
  }
 
  // Budget mode — pick the LARGEST single bar size that fits at qty ≥ 1.
  const budgetSgd = inputValue;
  const candidates = standardProducts
    .map((p) => {
      const tier = p.tiers.find((t) => 1 >= t.min_qty && (t.max_qty === null || 1 <= t.max_qty))
                ?? [...p.tiers].sort((a, b) => b.min_qty - a.min_qty)[0];
      if (!tier) return null;
      const priceEach = spotSgd * p.grams * (1 + tier.premium_pct / 100);
      if (priceEach > budgetSgd) return null;
      const qty = Math.floor(budgetSgd / priceEach);
      return { product: p, qty };
    })

/** Look up a standard-bar tier by quantity. Returns the matching tier
 *  or null if the quantity isn't covered (shouldn't happen — tiers
 *  should always cover qty>=1, but defensive). */
function findStandardTier(
  product: StandardProductForAI,
  quantity: number,
): { premium_pct: number } | null {
  for (const t of product.tiers) {
    if (quantity >= t.min_qty && (t.max_qty === null || quantity <= t.max_qty)) {
      return { premium_pct: t.premium_pct };
    }
  }
  // Fall back to the highest-min-qty tier (large-volume case)
  const sorted = [...product.tiers].sort((a, b) => b.min_qty - a.min_qty);
  return sorted[0] ? { premium_pct: sorted[0].premium_pct } : null;
}
