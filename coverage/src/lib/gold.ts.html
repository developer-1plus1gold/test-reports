// Ordered largest → smallest for the greedy breakdown algorithm
export const SIZES = [...PRODUCT_CATALOG]
  .sort((a, b) => b.g - a.g)
  .map((p) => ({ g: p.g, label: p.label, isAssay: p.isAssay }));
 
/**
 * Log-space interpolated premium for a given total cart weight.
 *
 * Anchors are the DB product premiums (Map<grams → premium_decimal>).
 * Property: more grams → equal or lower premium (monotonically non-increasing).
 *
 * - At or below the smallest anchor: return that anchor's premium (highest rate).
 * - At or above the largest anchor:  return that anchor's premium (floor rate).
 * - Between anchors: linear interpolation in log(grams) space.
 *
 * Using log-space keeps the curve smooth and natural for gold sizing
 * (products span 1g → 100g, a 100× range — log space spreads them evenly).
 */
export function getPremiumForWeight(
  totalGrams: number,
  anchors: Map<number, number>
): number {
  if (totalGrams <= 0 || anchors.size === 0) return 0;
 
  const sorted = [...anchors.entries()].sort((a, b) => a[0] - b[0]);
 
  // At or below smallest anchor — use smallest (highest) premium
  if (totalGrams <= sorted[0][0]) return sorted[0][1];
 
  // At or above largest anchor — use floor (lowest) premium
  if (totalGrams >= sorted[sorted.length - 1][0]) return sorted[sorted.length - 1][1];
 
  // Bracket search + log-space interpolation
  for (let i = 0; i < sorted.length - 1; i++) {
    const [q0, p0] = sorted[i];
    const [q1, p1] = sorted[i + 1];
    if (totalGrams >= q0 && totalGrams < q1) {
      const t = (Math.log(totalGrams) - Math.log(q0)) / (Math.log(q1) - Math.log(q0));
      return p0 + t * (p1 - p0);
    }
  }
 
  return sorted[sorted.length - 1][1]; // unreachable but safe fallback
}
 
// Each 50g cycle completes a premium curve reset. Customers continue from
// where they left off across orders — they never restart at 1g unless they
// complete a full 50g cycle.
const CURVE_MILESTONE_G = 50;
 
/**
 * Premium rate the customer's next 1g purchase will be charged.
 *
 * Mirrors what computeOrderTotal computes for a 1g cart: rate at the
 * effective weight after one more gram lands, capped at the cycle floor.
 * Use this anywhere a UI says "your next purchase rate" — dashboard tile,
 * /our-pricing customer dot, slider's default-pinned label — so all those
 * surfaces match the price the cart will actually quote.
 *
 * Distinct from `getPremiumForWeight(positionInCycle, anchors)` which
 * answers "what's the curve's value at exactly this X" — that's the
 * rate the customer has *achieved*, not what their next gram lands on.
 */
export function nextPurchaseRate(
  positionInCycle: number,
  anchors: Map<number, number>,
): number {
  const effectiveWeight = Math.min(positionInCycle + 1, CURVE_MILESTONE_G);
  return getPremiumForWeight(effectiveWeight, anchors);
}
 
/**
 * Canonical gold order pricing formula.
 *
 * This is the ONLY function that should be used to convert (grams, spot, premiumMap)
 * into a SGD total. Every order, payment, and display across the codebase must call
 * this — never inline `spot × grams × (1 + getPremiumForWeight(...))` directly.
 *
 * Premium is determined by the customer's effective curve position — their accumulated
 * curve grams from all paid orders (mod 50g) plus the current cart. This means a
 * customer who bought 30g previously continues from 30g, not 1g.
 *
 * Cap at 50g: if accumulated + cart crosses a 50g milestone, the entire cart gets
 * the 50g floor rate (best possible rate in the cycle).
 *
 * @param accumulatedCurveGrams - total curve grams from the customer's paid order
 *   history (mod 50g boundary is applied internally). Pass 0 for guests.
 *
 * @returns `{ totalSgd, curveRate }`
 *   `curveRate` is exposed so callers can derive per-item prices that sum correctly:
 *   `perItemPriceSgd = spot × itemGrams × (1 + curveRate)`
 */
export function computeOrderTotal(
  totalGrams: number,
  spotSgd: number,
  premiumMap: Map<number, number>,
  accumulatedCurveGrams = 0,
): { totalSgd: number; curveRate: number } {
  const positionInCycle = accumulatedCurveGrams % CURVE_MILESTONE_G;
  const effectiveWeight = Math.min(positionInCycle + totalGrams, CURVE_MILESTONE_G);
  const curveRate = getPremiumForWeight(effectiveWeight, premiumMap);
  return {
    totalSgd: spotSgd * totalGrams * (1 + curveRate),
    curveRate,
  };
}

export interface StandardPremiumTier {
  min_qty:     number;
  max_qty:     number | null;
  premium_pct: number;
}
 
/**
 * Returns the premium decimal (e.g. 0.10 for 10%) for a standard product
 * at the given quantity, based on its tier table.
 *
 * Tiers are non-overlapping qty ranges. If no tier matches (misconfigured data),
 * returns 0 so the product is still purchasable at spot with no markup.
 */
export function getTierPremium(qty: number, tiers: StandardPremiumTier[]): number {
  const match = tiers.find(
    (t) => qty >= t.min_qty && (t.max_qty === null || qty <= t.max_qty)
  );
  return match ? match.premium_pct / 100 : 0;
}
 
/**
 * Price a single standard product line item.
 *
 * Formula: spot × g_per_unit × qty × (1 + tier_premium(qty))
 * The tier is determined by THIS line's qty only — independent of the rest of the cart.
 *
 * @returns { lineSgd, priceEach, premiumRate }
 *   priceEach  — per-unit price at this qty (changes if qty crosses a tier boundary)
 *   premiumRate — the exact tier rate applied (stored on order_items.premium_applied_pct)
 */
export function computeStandardLineTotal(
  qty:        number,
  gPerUnit:   number,
  spotSgd:    number,
  tiers:      StandardPremiumTier[],
): { lineSgd: number; priceEach: number; premiumRate: number } {
  const premiumRate = getTierPremium(qty, tiers);
  const priceEach   = spotSgd * gPerUnit * (1 + premiumRate);
  return { lineSgd: priceEach * qty, priceEach, premiumRate };
}

export interface AccessoryDiscountRule {
  type:           "flat_pct" | "free";
  value?:         number;   // flat_pct: percentage (e.g. 20 = 20% off)
  threshold_sgd?: number;   // minimum gold subtotal to trigger; omit = always applies
  free_qty?:      number;   // free type only: max units that are free (default 1)
}
 
/**
 * Price a single accessory line item.
 *
 * Rules are sorted by threshold_sgd descending (highest = best deal = evaluated first),
 * then unconditional rules last. First matching rule wins.
 *
 * @param goldSubtotalSgd — the curve + standard gold subtotal in the cart (for threshold checks)
 */
export function computeAccessoryLineTotal(
  qty:              number,
  priceSgd:         number,
  discountRules:    AccessoryDiscountRule[] | null,
  goldSubtotalSgd:  number,
): {
  lineSgd:        number;
  freeQty:        number;  // units that cost $0 due to a free rule
  paidQty:        number;  // units charged at paidPriceEach
  paidPriceEach:  number;  // unit price for charged units (may be discounted by flat_pct)
  discountApplied: boolean;
} {
  let freeQty = 0;
  let paidPriceEach = priceSgd;
 
  if (discountRules && discountRules.length > 0) {
    // Sort: threshold rules highest-first, unconditional rules last
    const sorted = [
      ...discountRules.filter(r => r.threshold_sgd !== undefined).sort((a, b) => (b.threshold_sgd ?? 0) - (a.threshold_sgd ?? 0)),
      ...discountRules.filter(r => r.threshold_sgd === undefined),
    ];
 
    for (const rule of sorted) {
      if (rule.threshold_sgd !== undefined && goldSubtotalSgd < rule.threshold_sgd) continue;
 
      if (rule.type === "free") {
        freeQty = Math.min(qty, rule.free_qty ?? 1);
        break;
      } else if (Erule.type === "flat_pct" && rule.value !== undefined) {
        paidPriceEach = priceSgd * (1 - rule.value / 100);
        break;
      }
    }
  }
 
  const paidQty = qty - freeQty;
  const lineSgd = paidQty * paidPriceEach;
 
  return { lineSgd, freeQty, paidQty, paidPriceEach, discountApplied: freeQty > 0 || paidPriceEach < priceSgd };
}
 
/** Mutable version of GoldSize used in breakdown results (premium may be overridden from DB) */
export interface GoldSizeResolved {
  g: number;
  label: string;
  isAssay: boolean;
  premium: number;
}
 
/**
 * Curve-aware budget breakdown.
 *
 * Finds the maximum whole-gram count G such that:
 *   G × spot × (1 + curveRate(G)) ≤ budget
 *
 * This is the correct formulation because the continuous premium curve applies
 * to the ENTIRE order, not individual pieces. The old greedy approach priced
 * each gram at its per-product anchor (e.g. 20% for 1g), causing far fewer grams
 * to be bought than the budget could afford — a customer spending $5,000 might get
 * 21g when they could actually get 23g at the correct curve rate.
 *
 * Binary search works because cost(G) is monotonically increasing in G
 * (adding a gram always increases total spend, even though the unit rate drops).
 *
 * @param premiumOverrides - Map<grams, premium_decimal> from DB — used as curve anchors
 */
export function calculateBreakdown(
  budgetSgd: number,
  spotSgd: number,
  premiumOverrides?: Map<number, number>
): { size: GoldSizeResolved; count: number; priceEach: number; totalSgd: number }[] {
  if (!premiumOverrides || premiumOverrides.size === 0 || spotSgd <= 0 || budgetSgd <= 0) return [];
 
  // Upper bound: assume floor rate (lowest premium) so we never search past the true max
  const minPremium = Math.min(...premiumOverrides.values());
  const upper = Math.ceil(budgetSgd / (spotSgd * (1 + minPremium))) + 1;
 
  // Binary search for maximum affordable grams
  let lo = 0, hi = upper;
  while (lo < hi) {
    const mid = Math.floor((lo + hi + 1) / 2);
    const rate = getPremiumForWeight(mid, premiumOverrides);
    if (spotSgd * mid * (1 + rate) <= budgetSgd) {
      lo = mid;
    } else {
      hi = mid - 1;
    }
  }
 
  if (lo === 0) return [];
 
  const totalGrams = lo;
  const curveRate  = getPremiumForWeight(totalGrams, premiumOverrides);
 
  // Denomination split (greedy largest-first), priced at the unified curve rate
  const raw = calculateBreakdownByGrams(totalGrams, spotSgd, premiumOverrides);
  return raw.map((b) => ({
    ...b,
    priceEach: spotSgd * b.size.g * (1 + curveRate),
    totalSgd:  spotSgd * b.size.g * b.count * (1 + curveRate),
  }));
}
 
/**
 * Gram-target breakdown.
 * Given an exact gram target, greedily fills with the largest denominations first.
 * Returns the denomination mix and total cost — the user is buying exactly this many grams.
 */
export function calculateBreakdownByGrams(
  targetGrams: number,
  spotSgd: number,
  premiumOverrides?: Map<number, number>
): { size: GoldSizeResolved; count: number; priceEach: number; totalSgd: number }[] {
  const result: { size: GoldSizeResolved; count: number; priceEach: number; totalSgd: number }[] = [];
  let remaining = targetGrams;
 
  for (const size of SIZES) {
    const premium = premiumOverrides?.get(size.g);
    if (premium === undefined) continue;
    const count = Math.floor(remaining / size.g);
    if (count > 0) {
      const priceEach = spotSgd * size.g * (1 + premium);
      result.push({ size: { g: size.g, label: size.label, isAssay: size.isAssay, premium }, count, priceEach, totalSgd: count * priceEach });
      remaining = parseFloat((remaining - count * size.g).toFixed(6));
    }
  }
 
  return result;
}
 
/**
 * Returns the curve-based effective premium for a given breakdown.
 * Delegates to getPremiumForWeight — the premium is determined by total cart
 * weight, not a weighted average of per-product rates.
 *
 * If no anchors are provided (e.g. in tests), falls back to the per-product
 * weighted average so existing test fixtures don't need to be rewritten.
 */
export function blendedPremium(
  breakdown: { size: GoldSizeResolved; count: number; priceEach: number; totalSgd: number }[],
  anchors?: Map<number, number>
): number {
  const totalGrams = breakdown.reduce((s, b) => s + b.size.g * b.count, 0);
  if (totalGrams === 0) return 0;
  if (anchors) return getPremiumForWeight(totalGrams, anchors);
  // Test-only fallback: weighted average (no anchor map available)
  return breakdown.reduce((s, b) => s + b.size.premium * (b.size.g * b.count) / totalGrams, 0);
}
