/**
 * Replicate (Flux Pro) image generation wrapper.
 *
 * Thin client — submits a prediction, polls for completion, returns
 * the public output URL. MarketingGenerator agent owns the workflow
 * orchestration; this file is just the HTTP surface.
 *
 * Model: black-forest-labs/flux-1.1-pro (default). Override per call
 * via input.model.
 *
 * Pricing: ~$0.04 per image (1024×1024, single output). Tracked into
 * marketing_assets.generation_cost_usd at the call site.
 *
 * Env: REPLICATE_API_TOKEN — Mukesh provisions; this file expects it set.
 */
 
interface ReplicateInput {
  prompt:           string;
  negative_prompt?: string;
  aspect_ratio?:    "1:1" | "9:16" | "16:9" | "3:4" | "4:3";
  num_outputs?:     number;
  seed?:            number;
  output_format?:   "png" | "webp" | "jpg";
}
 
export interface GenerateImageRequest {
  prompt:           string;
  negative_prompt?: string;
  aspect_ratio?:    "1:1" | "9:16" | "16:9" | "3:4" | "4:3";
  /** Defaults to flux-1.1-pro. Override for cheaper/faster models. */
  model?:           string;
  seed?:            number;
  outputFormat?:    "png" | "webp" | "jpg";
}
 
export interface GenerateImageResult {
  /** Public Replicate URL of the generated image (24h TTL — fetch + persist). */
  imageUrl:    string;
  /** Replicate's prediction ID — kept for audit. */
  predictionId: string;
  /** Tracked cost — Replicate doesn't report it inline; we use the
   *  documented per-model rate. */
  costUsd:     number;
  /** Model actually used. */
  model:       string;
  /** Seed used (Replicate returns it even if unspecified). */
  seed:        number | null;
}
 
const REPLICATE_API = "https://api.replicate.com/v1";
const DEFAULT_MODEL = "black-forest-labs/flux-1.1-pro";
 
/** Per-model price per output ($USD). Update as Replicate adjusts. */
const PRICE_PER_OUTPUT_USD: Record<string, number> = {
  "black-forest-labs/flux-1.1-pro":      0.04,
  "black-forest-labs/flux-1.1-pro-ultra": 0.06,
  "black-forest-labs/flux-schnell":      0.003,
};
 
export async function generateImage(req: GenerateImageRequest): Promise<GenerateImageResult> {
  const token = process.env.REPLICATE_API_TOKEN;
  if (!token) throw new Error("REPLICATE_API_TOKEN not set");
 
  const model = req.model ?? DEFAULT_MODEL;
  const input: ReplicateInput = {
    prompt:           req.prompt,
    negative_prompt:  req.negative_prompt,
    aspect_ratio:     req.aspect_ratio ?? "1:1",
    num_outputs:      1,
    seed:             req.seed,
    output_format:    req.outputFormat ?? "webp",
  };
 
  // 1. Create prediction.
  const createR = await fetch(`${REPLICATE_API}/models/${model}/predictions`, {
    method: "POST",
    headers: {
      Authorization: `Bearer ${token}`,
      "Content-Type": "application/json",
      // Block until completion, max 60s. Past that we fall back to polling.
      Prefer: "wait=60",
    },
    body: JSON.stringify({ input }),
  });
  if (!createR.ok) {
    throw new Error(`Replicate create failed: HTTP ${createR.status} ${await createR.text().catch(()=>"")}`);
  }
  let pred = await createR.json() as { id: string; status: string; output: string | string[] | null; error?: string };
 
  // 2. Poll until terminal state if not already done via Prefer:wait.
  const POLL_MS  = 1500;
  const TIMEOUT  = Date.now() + 120_000;
  while (pred.status !== "succeeded" && pred.status !== "failed" && pred.status !== "canceled") {
    if (Date.now() > TIMEOUT) throw new Error(`Replicate prediction ${pred.id} timed out after 120s`);
    await new Promise(r => setTimeout(r, POLL_MS));
    const pollR = await fetch(`${REPLICATE_API}/predictions/${pred.id}`, {
      headers: { Authorization: `Bearer ${token}` },
    });
    if (!pollR.ok) throw new Error(`Replicate poll failed: HTTP ${pollR.status}`);
    pred = await pollR.json();
  }
 
  if (pred.status !== "succeeded") {
    throw new Error(`Replicate prediction ${pred.id} ended with status=${pred.status}${pred.error ? ` error=${pred.error}` : ""}`);
  }
 
  // Output can be a single string or array depending on num_outputs.
  const imageUrl = Array.isArray(pred.output) ? pred.output[0] : pred.output;
  if (!imageUrl) throw new Error(`Replicate prediction ${pred.id} succeeded but output empty`);
 
  return {
    imageUrl,
    predictionId: pred.id,
    costUsd:      PRICE_PER_OUTPUT_USD[model] ?? 0.04,
    model,
    seed:         req.seed ?? null,
  };
}
