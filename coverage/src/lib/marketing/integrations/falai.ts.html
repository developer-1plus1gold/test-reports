/**
 * fal.ai (Kling) video generation wrapper.
 *
 * Calls the Kling text-to-video or image-to-video endpoint via fal's
 * queue API. Polls for completion; returns the MP4 URL.
 *
 * Pricing: ~$0.40 per 5s clip (Kling 1.5 Standard) or ~$0.80 per 5s
 * (Kling 1.6 Pro). Tracked into marketing_assets.generation_cost_usd.
 *
 * Env: FAL_KEY — Mukesh provisions.
 */
 
const FAL_API = "https://fal.run";
 
export interface GenerateVideoRequest {
  prompt:        string;
  /** Optional reference image URL for image-to-video. Kling supports up
   *  to one reference image. */
  imageUrl?:     string;
  /** Defaults to fal-ai/kling-video/v1.6/pro. */
  model?:        string;
  /** Clip duration in seconds. Kling supports 5 or 10. */
  durationSec?:  5 | 10;
  /** Aspect ratio. Kling supports 16:9 / 9:16 / 1:1. */
  aspectRatio?: "16:9" | "9:16" | "1:1";
  /** Optional creative seed. */
  seed?:         number;
}
 
export interface GenerateVideoResult {
  videoUrl:     string;
  requestId:    string;
  costUsd:      number;
  model:        string;
  durationSec:  number;
}
 
const PRICE_PER_5S_CLIP_USD: Record<string, number> = {
  "fal-ai/kling-video/v1.6/pro":           0.80,
  "fal-ai/kling-video/v1.5/pro":           0.70,
  "fal-ai/kling-video/v1.5/standard":      0.40,
};
 
export async function generateVideo(req: GenerateVideoRequest): Promise<GenerateVideoResult> {
  const token = process.env.FAL_KEY;
  if (!token) throw new Error("FAL_KEY not set");
 
  const model        = req.model ?? "fal-ai/kling-video/v1.6/pro";
  const durationSec  = req.durationSec ?? 5;
 
  // fal.ai split text-to-video and image-to-video into separate endpoint
  // paths under each model. The model identifier we accept names the family
  // (e.g. "fal-ai/kling-video/v1.6/pro"); we append the variant here based
  // on whether a reference image was supplied.
  const variant   = req.imageUrl ? "image-to-video" : "text-to-video";
  const modelPath = `${model}/${variant}`;
 
  // fal.ai accepts sync calls via /run that block up to 3min. Kling clips
  // typically take 60-120s. If we time out, the route returns 408 and we
  // fall back to async queue polling — but for now sync is fine.
  const r = await fetch(`${FAL_API}/${modelPath}`, {
    method: "POST",
    headers: {
      Authorization: `Key ${token}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({
      prompt:        req.prompt,
      image_url:     req.imageUrl,
      duration:      `${durationSec}`,
      aspect_ratio:  req.aspectRatio ?? "9:16",
      seed:          req.seed,
    }),
  });
  if (!r.ok) {
    throw new Error(`fal.ai generate failed: HTTP ${r.status} ${await r.text().catch(()=>"")}`);
  }
  const result = await r.json() as { video?: { url: string }; request_id?: string };
  const videoUrl = result.video?.url;
  if (!videoUrl) throw new Error(`fal.ai response missing video URL: ${JSON.stringify(result).slice(0, 300)}`);
 
  // Cost scales linearly with duration (5s -> base; 10s -> 2× base).
  const baseCost = PRICE_PER_5S_CLIP_USD[model] ?? 0.80;
  const costUsd = baseCost * (durationSec / 5);
 
  return {
    videoUrl,
    requestId:   result.request_id ?? "unknown",
    costUsd,
    model,
    durationSec,
  };
}
