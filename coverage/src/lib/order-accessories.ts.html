/**
 * Shared helper: compute and insert accessory order_items after an order is created.
 *
 * Accessories have grams_each=0 and don't affect gold totals, accumulation, or stock.
 * Their price_sgd is 0 for free accessories or the discounted rate for flat_pct rules.
 * Non-fatal — if the insert fails the order still exists; accessories are bonus items.
 */
 
import type { SupabaseClient } from "@supabase/supabase-js";
import { computeAccessoryLineTotal } from "@/lib/gold";
 
export interface AccessoryLineItem {
  productId: string;
  quantity:  number;
}
