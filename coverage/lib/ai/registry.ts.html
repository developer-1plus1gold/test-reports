/**
 * Model-tier registry — resolves a tier name to a concrete model + pricing.
 *
 * Reads from the `ai_model_config` DB table so the admin UI can swap models
 * without a redeploy. Reads are cached for 60 seconds to amortise DB cost
 * across the busy bursts of inferences a single page-load can trigger.
 *
 * Tiers (seeded by 20260429_ai_phase0_registries.sql):
 *   cheap     → Claude Haiku 4.5 — default for low-stakes work
 *   fast      → same as cheap; alias kept for callers expressing intent
 *   reasoning → Claude Sonnet 4.6 — compliance + decisions with reg weight
 *   hard      → Claude Opus 4.7 — highest-stakes single-shot reasoning
 *
 * Adding a tier or rewiring an existing tier is a row-edit in the DB; no
 * code change required. The registry returns a frozen snapshot the caller
 * shouldn't mutate.
 */
 
import { createClient } from "@supabase/supabase-js";
 
export type ModelTier = "cheap" | "fast" | "reasoning" | "hard";
 
export interface ResolvedModel {
  tier:                     ModelTier;
  modelId:                  string;
  displayName:              string;
  inputPerMtokUsd:          number;
  outputPerMtokUsd:         number;
  cacheReadPerMtokUsd:      number;
  cacheWritePerMtokUsd:     number;
  capabilityTags:           string[];
}
 
const CACHE_TTL_MS = 60_000;
let _cache: { fetchedAt: number; tiers: Map<ModelTier, ResolvedModel> } | null = null;
 
// eslint-disable-next-line @typescript-eslint/no-explicit-any
let _supabase: any = null;
function admin() {
  if (_supabase) return _supabase;
  _supabase = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.SUPABASE_SERVICE_KEY!,
  );
  return _supabase;
}
 
async function loadAll(): Promise<Map<ModelTier, ResolvedModel>> {
  const { data, error } = await admin()
    .from("ai_model_config")
    .select("tier, model_id, display_name, input_per_mtok_usd, output_per_mtok_usd, cache_read_per_mtok_usd, cache_write_per_mtok_usd, capability_tags, enabled");
 
  if (error) {
    throw new Error(`ai_model_config read failed: ${error.message}`);
  }
 
  const m = new Map<ModelTier, ResolvedModel>();
  for (const row of (data ?? []) as Array<{
    tier: string; model_id: string; display_name: string;
    input_per_mtok_usd: number | null; output_per_mtok_usd: number | null;
    cache_read_per_mtok_usd: number | null; cache_write_per_mtok_usd: number | null;
    capability_tags: string[] | null; enabled: boolean;
  }>) {
    if (!row.enabled) continue;
    m.set(row.tier as ModelTier, {
      tier:                  row.tier as ModelTier,
      modelId:               row.model_id,
      displayName:           row.display_name,
      inputPerMtokUsd:       Number(row.input_per_mtok_usd ?? 0),
      outputPerMtokUsd:      Number(row.output_per_mtok_usd ?? 0),
      cacheReadPerMtokUsd:   Number(row.cache_read_per_mtok_usd ?? 0),
      cacheWritePerMtokUsd:  Number(row.cache_write_per_mtok_usd ?? 0),
      capabilityTags:        row.capability_tags ?? [],
    });
  }
  return m;
}
 
/**
 * Returns the live model resolution for a tier. Cached for ~60s.
 * Throws if the tier is missing or the DB read fails — agents should
 * never silently fall back to a hardcoded model.
 */
export async function resolveTier(tier: ModelTier): Promise<ResolvedModel> {
  const now = Date.now();
  if (!_cache || now - _cache.fetchedAt > CACHE_TTL_MS) {
    _cache = { fetchedAt: now, tiers: await loadAll() };
  }
  const resolved = _cache.tiers.get(tier);
  if (!resolved) {
    throw new Error(`No active model configured for tier "${tier}"`);
  }
  return resolved;
}
 
/**
 * Compute the USD cost of a single inference given the model's pricing
 * and the token usage Anthropic reported.
 */
export function calcCostUsd(
  model:     ResolvedModel,
  usage:     { input?: number; output?: number; cacheRead?: number; cacheWrite?: number },
): number {
  const i  = (usage.input      ?? 0) * model.inputPerMtokUsd      / 1_000_000;
  const o  = (usage.output     ?? 0) * model.outputPerMtokUsd     / 1_000_000;
  const cr = (usage.cacheRead  ?? 0) * model.cacheReadPerMtokUsd  / 1_000_000;
  const cw = (usage.cacheWrite ?? 0) * model.cacheWritePerMtokUsd / 1_000_000;
  return Math.round((i + o + cr + cw) * 1_000_000) / 1_000_000;
}
 
/** For tests — drops the in-memory cache so the next call hits the DB. */
export function _resetRegistryCacheForTesting(): void {
  _cache = null;
}
